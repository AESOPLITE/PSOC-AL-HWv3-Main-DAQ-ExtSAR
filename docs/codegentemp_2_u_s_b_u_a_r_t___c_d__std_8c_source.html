481 }
482
483 #if (!USBUART_CD_EP_MANAGEMENT_DMA_AUTO)

844 USBUART_CD_EP[curEp].attrib = pEP->attributes;

856 USBUART_CD_EP[curEp].epMode = USBUART_CD_GET_ACTIVE_OUT_EP_CR0_MODE(epType);
857 USBUART_CD_EP[curEp].apiEpState = USBUART_CD_NO_EVENT_PENDING;

861 USBUART_CD_SIE_EP_BASE.sieEp[curEp].epCr0 = USBUART_CD_MODE_NAK_IN_OUT;

866

869

876 }

903 */
904 USBUART_CD_DYN_RECONFIG_REG &= (uint8) ~USBUART_CD_DYN_RECONFIG_ENABLE;
905
906 #else

908 USBUART_CD_SIE_EP_BASE.sieEp[curEp].epCnt1 = LO8(USBUART_CD_EP[curEp].bufferSize);
909
910 #if (CY_PSOC4)
911 USBUART_CD_ARB_EP16_BASE.arbEp[curEp].rwRa16 = (uint32) USBUART_CD_EP[curEp].buffOffset;
912 USBUART_CD_ARB_EP16_BASE.arbEp[curEp].rwWa16 = (uint32) USBUART_CD_EP[curEp].buffOffset;
913 #else
914 USBUART_CD_ARB_EP_BASE.arbEp[curEp].rwRa = LO8(USBUART_CD_EP[curEp].buffOffset);
915 USBUART_CD_ARB_EP_BASE.arbEp[curEp].rwRaMsb = HI8(USBUART_CD_EP[curEp].buffOffset);
916 USBUART_CD_ARB_EP_BASE.arbEp[curEp].rwWa = LO8(USBUART_CD_EP[curEp].buffOffset);
917 USBUART_CD_ARB_EP_BASE.arbEp[curEp].rwWaMsb = HI8(USBUART_CD_EP[curEp].buffOffset);
918 #endif /* (CY_PSOC4) */
919 #endif /* (USBUART_CD_EP_MANAGEMENT_DMA_AUTO) */
920 }

922 pEP = &pEP[1u]; /* Get next EP element */
923 }
924
925 /* The main loop has to re-enable DMA and OUT endpoint */
926 }
