140cystatus CyFlash_EraseRow(uint8 arrayId, uint16 rowAddress)
141{
142 cystatus status = CYRET_SUCCESS;
143
144 if (arrayId > CY_SPC_LAST_FLASH_ARRAYID)
145 {
146 status = CYRET_BAD_PARAM;
147 }
148 else if(rowAddress > (CY_FLASH_NUMBER_ROWS/CY_FLASH_NUMBER_ARRAYS))
149 {
150 status = CYRET_BAD_PARAM;
151 }
152 else
153 {
154 if(CySpcLock() == CYRET_SUCCESS)
155 {
156 /* Erase flash row */
157 status = CySpcEraseRow(arrayId, rowAddress, dieTemperature[0u], dieTemperature[1u]);
158
159 if(CYRET_STARTED == status)
160 {
161 while(CY_SPC_BUSY)
162 {
163 /* Wait for SPC to finish and get SPC status */
164 CyDelayUs(1u);
165 }
166
167 /* Hide SPC status */
168 if(CY_SPC_STATUS_SUCCESS == CY_SPC_READ_STATUS)
169 {
170 status = CYRET_SUCCESS;
171 }
172 else
173 {
174 status = CYRET_UNKNOWN;
175 }
176 }
177 else
178 {
179 status = CYRET_CANCELED;
180 }
181 CySpcUnlock();
182 }
183 else
184 {
185 status = CYRET_LOCKED;
186 }
187 }
188
189 return(status);
190}

208static cystatus CySetTempInt(void)
209{
210 cystatus status;
211
212 /* Make sure SPC is powered */
213 CySpcStart();
214
215 /* Plan for failure. */
216 status = CYRET_UNKNOWN;
217
218 if(CySpcLock() == CYRET_SUCCESS)
219 {
220 /* Write the command. */
221 if(CYRET_STARTED == CySpcGetTemp(CY_TEMP_NUMBER_OF_SAMPLES))
222 {
223 do
224 {
225 if(CySpcReadData(dieTemperature, CY_FLASH_DIE_TEMP_DATA_SIZE) == CY_FLASH_DIE_TEMP_DATA_SIZE)
226 {
227 status = CYRET_SUCCESS;
228
229 while(CY_SPC_BUSY)
230 {
231 /* Spin until idle. */
232 CyDelayUs(1u);
233 }
234 break;
235 }
236
237 } while(CY_SPC_BUSY);
238 }
239
240 CySpcUnlock();
241 }
242 else
243 {
244 status = CYRET_LOCKED;

480cystatus CyWriteRowFull(uint8 arrayId, uint16 rowNumber, const uint8* rowData, uint16 rowSize) \
481
482{
483 cystatus status = CYRET_SUCCESS;
484
485 if((arrayId <= CY_SPC_LAST_FLASH_ARRAYID) && (arrayId > (CY_FLASH_NUMBER_ARRAYS + CY_SPC_FIRST_FLASH_ARRAYID)))
486 {
487 status = CYRET_BAD_PARAM;
488 }
489
490 if(arrayId > CY_SPC_LAST_EE_ARRAYID)
491 {
492 status = CYRET_BAD_PARAM;
493 }
494
495 if((arrayId >= CY_SPC_FIRST_EE_ARRAYID) && (arrayId > (CY_FLASH_EEPROM_NUMBER_ARRAYS + CY_SPC_FIRST_EE_ARRAYID)))
496 {
497 status = CYRET_BAD_PARAM;
498 }
499
500 if(arrayId <= CY_SPC_LAST_FLASH_ARRAYID)
501 {
502 /* Flash */
503 if(rowNumber > (CY_FLASH_NUMBER_ROWS/CY_FLASH_NUMBER_ARRAYS))
504 {
505 status = CYRET_BAD_PARAM;
506 }
507 }
508 else
509 {
510 /* EEPROM */
511 if(rowNumber > (CY_EEPROM_NUMBER_ROWS/CY_FLASH_EEPROM_NUMBER_ARRAYS))
512 {
513 status = CYRET_BAD_PARAM;
514 }
515
516 if(CY_EEPROM_SIZEOF_ROW != rowSize)
517 {
518 status = CYRET_BAD_PARAM;
519 }
520 }
521
522 if(rowData == NULL)
523 {
524 status = CYRET_BAD_PARAM;
525 }
526
527
528 if(status == CYRET_SUCCESS)
529 {
530 if(CySpcLock() == CYRET_SUCCESS)
531 {
532 /* Load row data into SPC internal latch */
533 status = CySpcLoadRowFull(arrayId, rowNumber, rowData, rowSize);
534
535 if(CYRET_STARTED == status)
536 {
537 while(CY_SPC_BUSY)
538 {
539 /* Wait for SPC to finish and get SPC status */
540 CyDelayUs(1u);
541 }
542
543 /* Hide SPC status */
544 if(CY_SPC_STATUS_SUCCESS == CY_SPC_READ_STATUS)
545 {
546 status = CYRET_SUCCESS;
547 }
548 else
549 {
550 status = CYRET_UNKNOWN;
551 }
552
553 if(CYRET_SUCCESS == status)
554 {
555 /* Erase and program flash with data from SPC interval latch */
556 status = CySpcWriteRow(arrayId, rowNumber, dieTemperature[0u], dieTemperature[1u]);
557
558 if(CYRET_STARTED == status)
559 {
560 while(CY_SPC_BUSY)
561 {
562 /* Wait for SPC to finish and get SPC status */
563 CyDelayUs(1u);
564 }
565
566 /* Hide SPC status */
567 if(CY_SPC_STATUS_SUCCESS == CY_SPC_READ_STATUS)
568 {
569 status = CYRET_SUCCESS;
570 }
571 else
572 {
573 status = CYRET_UNKNOWN;
574 }
575 }
576 }
577 }
578 CySpcUnlock();
579 } /* if(CySpcLock() == CYRET_SUCCESS) */
580 else
581 {
582 status = CYRET_LOCKED;
583 }
584 }
585
586 return(status);
587}

603void CyFlash_SetWaitCycles(uint8 freq)
604{
605 uint8 interruptState;
606
607 /* Save current global interrupt enable and disable it */
608 interruptState = CyEnterCriticalSection();
609
610
616 if (freq < CY_FLASH_CACHE_WS_1_FREQ_MAX)
617 {
618 CY_FLASH_CONTROL_REG = (CY_FLASH_CONTROL_REG & (uint8)(~CY_FLASH_CACHE_WS_VALUE_MASK)) |
619 CY_FLASH_CACHE_WS_1_VALUE_MASK;
620 }
621 else if (freq < CY_FLASH_CACHE_WS_2_FREQ_MAX)
622 {
623 CY_FLASH_CONTROL_REG = (CY_FLASH_CONTROL_REG & (uint8)(~CY_FLASH_CACHE_WS_VALUE_MASK)) |
624 CY_FLASH_CACHE_WS_2_VALUE_MASK;
625 }
626 else if (freq < CY_FLASH_CACHE_WS_3_FREQ_MAX)
627 {
628 CY_FLASH_CONTROL_REG = (CY_FLASH_CONTROL_REG & (uint8)(~CY_FLASH_CACHE_WS_VALUE_MASK)) |
629 CY_FLASH_CACHE_WS_3_VALUE_MASK;
630 }
631#if (CY_PSOC5)
632 else if (freq < CY_FLASH_CACHE_WS_4_FREQ_MAX)
633 {
634 CY_FLASH_CONTROL_REG = (CY_FLASH_CONTROL_REG & (uint8)(~CY_FLASH_CACHE_WS_VALUE_MASK)) |
635 CY_FLASH_CACHE_WS_4_VALUE_MASK;
636 }
637 else if (freq <= CY_FLASH_CACHE_WS_5_FREQ_MAX)
638 {
639 CY_FLASH_CONTROL_REG = (CY_FLASH_CONTROL_REG & (uint8)(~CY_FLASH_CACHE_WS_VALUE_MASK)) |
640 CY_FLASH_CACHE_WS_5_VALUE_MASK;
641 }
642#endif /* (CY_PSOC5) */
643 else
644 {
645 /* Halt CPU in debug mode if frequency is invalid */
646 CYASSERT(0u != 0u);
647 }
648
649 /* Restore global interrupt enable state */
650 CyExitCriticalSection(interruptState);
