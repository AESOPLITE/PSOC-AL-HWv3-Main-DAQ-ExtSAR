53void CyDmacConfigure(void)
54{

163uint8 CyDmaChAlloc(void)
164{
165 uint8 interruptState;
166 uint8 dmaIndex;
167 uint32 channel = 1u;
168
169
170 /* Enter critical section! */
171 interruptState = CyEnterCriticalSection();
172
173 /* Look for free channel. */
174 for(dmaIndex = 0u; dmaIndex < CY_DMA_NUMBEROF_CHANNELS; dmaIndex++)
175 {
176 if(0uL == (CyDmaChannels & channel))
177 {
178 /* Mark channel as used. */
179 CyDmaChannels |= channel;
180 break;
181 }
182
183 channel <<= 1u;
184 }
185
186 if(dmaIndex >= CY_DMA_NUMBEROF_CHANNELS)
187 {
188 dmaIndex = CY_DMA_INVALID_CHANNEL;
189 }
190
191 /* Exit critical section! */
192 CyExitCriticalSection(interruptState);
193
194 return(dmaIndex);
195}

680{
681 uint8 interruptState;
682 uint8 element = CY_DMA_INVALID_TD;
683
684 /* Enter critical section! */
685 interruptState = CyEnterCriticalSection();
686
687 if(CyDmaTdCurrentNumber > NUMBEROF_CHANNELS)
