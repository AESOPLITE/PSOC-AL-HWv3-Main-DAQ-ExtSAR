775{
776 uint8 masterClkDiv;
777 uint16 busClkDiv;
778 uint8 interruptState;
779
780 interruptState = CyEnterCriticalSection();
781
782 /* Work around to set bus clock divider value */
783 busClkDiv = (uint16)((uint16)CY_LIB_CLKDIST_BCFG_MSB_REG << 8u);
784 busClkDiv |= CY_LIB_CLKDIST_BCFG_LSB_REG;
785
786 if ((divider == 0u) || (busClkDiv == 0u))
787 {
788 /* Save away master clock divider value */
789 masterClkDiv = CY_LIB_CLKDIST_MSTR0_REG;
790
791 if (masterClkDiv < CY_LIB_CLKDIST_MASTERCLK_DIV)
792 {
793 /* Set master clock divider to 7 */
794 CyMasterClk_SetDivider(CY_LIB_CLKDIST_MASTERCLK_DIV);
795 }
796
797 if (divider == 0u)
798 {
799 /* Set SSS bit and divider register desired value */
800 CY_LIB_CLKDIST_BCFG2_REG |= CY_LIB_CLKDIST_BCFG2_SSS;
801 CyBusClk_Internal_SetDivider(divider);

805 CyBusClk_Internal_SetDivider(divider);

810 CyMasterClk_SetDivider(masterClkDiv);
811 }
812 else

814 CyBusClk_Internal_SetDivider(divider);
815 }
