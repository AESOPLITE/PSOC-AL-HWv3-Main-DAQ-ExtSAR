1041 void UART_Cmd_PutChar(uint8 txDataByte)
1042 {
1043 #if (UART_Cmd_TX_INTERRUPT_ENABLED)
1044 /* The temporary output pointer is used since it takes two instructions
1045 * to increment with a wrap, and we can't risk doing that with the real
1046 * pointer and getting an interrupt in between instructions.
1047 */
1048 uint8 locTxBufferWrite;
1049 uint8 locTxBufferRead;
1050
1051 do
1052 { /* Block if software buffer is full, so we don't overwrite. */
1053
1054 #if ((UART_Cmd_TX_BUFFER_SIZE > UART_Cmd_MAX_BYTE_VALUE) && (CY_PSOC3))
1055 /* Disable TX interrupt to protect variables from modification */
1056 UART_Cmd_DisableTxInt();
1057 #endif /* (UART_Cmd_TX_BUFFER_SIZE > UART_Cmd_MAX_BYTE_VALUE) && (CY_PSOC3) */
1058
1059 locTxBufferWrite = UART_Cmd_txBufferWrite;
1060 locTxBufferRead = UART_Cmd_txBufferRead;
1061
1062 #if ((UART_Cmd_TX_BUFFER_SIZE > UART_Cmd_MAX_BYTE_VALUE) && (CY_PSOC3))
1063 /* Enable interrupt to continue transmission */
1064 UART_Cmd_EnableTxInt();
1065 #endif /* (UART_Cmd_TX_BUFFER_SIZE > UART_Cmd_MAX_BYTE_VALUE) && (CY_PSOC3) */
1066 }
1067 while( (locTxBufferWrite < locTxBufferRead) ? (locTxBufferWrite == (locTxBufferRead - 1u)) :
1068 ((locTxBufferWrite - locTxBufferRead) ==
1069 (uint8)(UART_Cmd_TX_BUFFER_SIZE - 1u)) );
1070
1071 if( (locTxBufferRead == locTxBufferWrite) &&
1072 ((UART_Cmd_TXSTATUS_REG & UART_Cmd_TX_STS_FIFO_FULL) == 0u) )
1073 {
1074 /* Add directly to the FIFO */
1075 UART_Cmd_TXDATA_REG = txDataByte;
1076 }
1077 else
1078 {
1079 if(locTxBufferWrite >= UART_Cmd_TX_BUFFER_SIZE)
1080 {
1081 locTxBufferWrite = 0u;
1082 }
1083 /* Add to the software buffer. */
1084 UART_Cmd_txBuffer[locTxBufferWrite] = txDataByte;
1085 locTxBufferWrite++;
1086
1087 /* Finally, update the real output pointer */
1088 #if ((UART_Cmd_TX_BUFFER_SIZE > UART_Cmd_MAX_BYTE_VALUE) && (CY_PSOC3))
1089 UART_Cmd_DisableTxInt();
1090 #endif /* (UART_Cmd_TX_BUFFER_SIZE > UART_Cmd_MAX_BYTE_VALUE) && (CY_PSOC3) */
1091
1092 UART_Cmd_txBufferWrite = locTxBufferWrite;
1093
1094 #if ((UART_Cmd_TX_BUFFER_SIZE > UART_Cmd_MAX_BYTE_VALUE) && (CY_PSOC3))
1095 UART_Cmd_EnableTxInt();
1096 #endif /* (UART_Cmd_TX_BUFFER_SIZE > UART_Cmd_MAX_BYTE_VALUE) && (CY_PSOC3) */
1097
1098 if(0u != (UART_Cmd_TXSTATUS_REG & UART_Cmd_TX_STS_FIFO_EMPTY))
1099 {
1100 /* Trigger TX interrupt to send software buffer */
1101 UART_Cmd_SetPendingTxInt();
1102 }
1103 }
1104
1105 #else
