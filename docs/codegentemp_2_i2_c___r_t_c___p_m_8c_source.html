72void I2C_RTC_SaveConfig(void)
73{
74#if (I2C_RTC_FF_IMPLEMENTED)
75 #if (I2C_RTC_WAKEUP_ENABLED)
76 uint8 intState;
77 #endif /* (I2C_RTC_WAKEUP_ENABLED) */
78
79 /* Store registers before enter low power mode */
80 I2C_RTC_backup.cfg = I2C_RTC_CFG_REG;
81 I2C_RTC_backup.xcfg = I2C_RTC_XCFG_REG;
82 I2C_RTC_backup.addr = I2C_RTC_ADDR_REG;
83 I2C_RTC_backup.clkDiv1 = I2C_RTC_CLKDIV1_REG;
84 I2C_RTC_backup.clkDiv2 = I2C_RTC_CLKDIV2_REG;
85
86#if (I2C_RTC_WAKEUP_ENABLED)
87 /* Disable master */
88 I2C_RTC_CFG_REG &= (uint8) ~I2C_RTC_ENABLE_MASTER;
89
90 /* Enable backup regulator to keep block powered in low power mode */
91 intState = CyEnterCriticalSection();
92 I2C_RTC_PWRSYS_CR1_REG |= I2C_RTC_PWRSYS_CR1_I2C_REG_BACKUP;
93 CyExitCriticalSection(intState);
94
95 /* 1) Set force NACK to ignore I2C transactions;
96 * 2) Wait unti I2C is ready go to Sleep; !!
97 * 3) These bits are cleared on wake up.
98 */
99 /* Wait when block is ready for sleep */
100 I2C_RTC_XCFG_REG |= I2C_RTC_XCFG_FORCE_NACK;
101 while (0u == (I2C_RTC_XCFG_REG & I2C_RTC_XCFG_RDY_TO_SLEEP))
102 {
103 }
104
105 /* Setup wakeup interrupt */
106 I2C_RTC_DisableInt();
107 (void) CyIntSetVector(I2C_RTC_ISR_NUMBER, &I2C_RTC_WAKEUP_ISR);
108 I2C_RTC_wakeupSource = 0u;
109 I2C_RTC_EnableInt();
110#endif /* (I2C_RTC_WAKEUP_ENABLED) */
111
112#else
113 /* Store only address match bit */
114 I2C_RTC_backup.control = (I2C_RTC_CFG_REG & I2C_RTC_CTRL_ANY_ADDRESS_MASK);
115#endif /* (I2C_RTC_FF_IMPLEMENTED) */
116
117#if (I2C_RTC_TIMEOUT_ENABLED)
118 I2C_RTC_TimeoutSaveConfig();
119#endif /* (I2C_RTC_TIMEOUT_ENABLED) */
120}

209void I2C_RTC_RestoreConfig(void)
210{
211#if (I2C_RTC_FF_IMPLEMENTED)
212 uint8 intState;
213
214 if (I2C_RTC_CHECK_PWRSYS_I2C_BACKUP)
215 /* Low power mode was Sleep - backup regulator is enabled */
216 {
217 /* Enable backup regulator in active mode */
218 intState = CyEnterCriticalSection();
219 I2C_RTC_PWRSYS_CR1_REG &= (uint8) ~I2C_RTC_PWRSYS_CR1_I2C_REG_BACKUP;
220 CyExitCriticalSection(intState);
221
222 /* Restore master */
223 I2C_RTC_CFG_REG = I2C_RTC_backup.cfg;
224 }
225 else
226 /* Low power mode was Hibernate - backup regulator is disabled. All registers are cleared */
227 {
228 #if (I2C_RTC_WAKEUP_ENABLED)
229 /* Disable power to block before register restore */
230 intState = CyEnterCriticalSection();
231 I2C_RTC_ACT_PWRMGR_REG &= (uint8) ~I2C_RTC_ACT_PWR_EN;
232 I2C_RTC_STBY_PWRMGR_REG &= (uint8) ~I2C_RTC_STBY_PWR_EN;
233 CyExitCriticalSection(intState);
234
235 /* Enable component in I2C_Wakeup() after register restore */
236 I2C_RTC_backup.enableState = I2C_RTC_ENABLE;
237 #endif /* (I2C_RTC_WAKEUP_ENABLED) */
238
239 /* Restore component registers after Hibernate */
240 I2C_RTC_XCFG_REG = I2C_RTC_backup.xcfg;
241 I2C_RTC_CFG_REG = I2C_RTC_backup.cfg;
242 I2C_RTC_ADDR_REG = I2C_RTC_backup.addr;
243 I2C_RTC_CLKDIV1_REG = I2C_RTC_backup.clkDiv1;
244 I2C_RTC_CLKDIV2_REG = I2C_RTC_backup.clkDiv2;
245 }
246
247#if (I2C_RTC_WAKEUP_ENABLED)
248 I2C_RTC_DisableInt();
249 (void) CyIntSetVector(I2C_RTC_ISR_NUMBER, &I2C_RTC_ISR);
250 if (0u != I2C_RTC_wakeupSource)
251 {
252 /* Generate interrupt to process incoming transaction */
253 I2C_RTC_SetPendingInt();
254 }
255 I2C_RTC_EnableInt();
256#endif /* (I2C_RTC_WAKEUP_ENABLED) */
257
258#else
259 I2C_RTC_CFG_REG = I2C_RTC_backup.control;
260#endif /* (I2C_RTC_FF_IMPLEMENTED) */
261
262#if (I2C_RTC_TIMEOUT_ENABLED)
263 I2C_RTC_TimeoutRestoreConfig();
264#endif /* (I2C_RTC_TIMEOUT_ENABLED) */
265}
