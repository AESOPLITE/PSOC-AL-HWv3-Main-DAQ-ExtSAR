646cystatus CySpcLock(void)
647{
648 cystatus status = CYRET_LOCKED;
649 uint8 interruptState;
650
651 /* Enter critical section */
652 interruptState = CyEnterCriticalSection();
653
654 if(CY_SPC_UNLOCKED == SpcLockState)
655 {
656 SpcLockState = CY_SPC_LOCKED;

664 CY_SPC_CPU_WAITPIPE_REG &= ((uint32)(~CY_SPC_CPU_WAITPIPE_BYPASS));
665

669 CY_NOP;
670
671 spcWaitPipeBypass = CY_SPC_CPU_WAITPIPE_BYPASS;
672 }
673
674 #endif /* (CY_PSOC5) */
675 }
676
677 /* Exit critical section */
678 CyExitCriticalSection(interruptState);
