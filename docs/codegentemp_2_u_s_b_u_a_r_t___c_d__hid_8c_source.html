76uint8 USBUART_CD_UpdateHIDTimer(uint8 interface)
77{
78 uint8 stat = USBUART_CD_IDLE_TIMER_INDEFINITE;
79
80 if(USBUART_CD_hidIdleRate[interface] != 0u)
81 {
82 if(USBUART_CD_hidIdleTimer[interface] > 0u)
83 {
84 USBUART_CD_hidIdleTimer[interface]--;
85 stat = USBUART_CD_IDLE_TIMER_RUNNING;
86 }

89 USBUART_CD_hidIdleTimer[interface] = USBUART_CD_hidIdleRate[interface];

91 }
92 }

131uint8 USBUART_CD_DispatchHIDClassRqst(void)
132{
133 uint8 requestHandled = USBUART_CD_FALSE;
134
135 uint8 interfaceNumber = (uint8) USBUART_CD_wIndexLoReg;
136
137 /* Check request direction: D2H or H2D. */
138 if (0u != (USBUART_CD_bmRequestTypeReg & USBUART_CD_RQST_DIR_D2H))
139 {
140 /* Handle direction from device to host. */
141
142 switch (USBUART_CD_bRequestReg)
143 {
144 case USBUART_CD_GET_DESCRIPTOR:
145 if (USBUART_CD_wValueHiReg == USBUART_CD_DESCR_HID_CLASS)
146 {
147 USBUART_CD_FindHidClassDecriptor();
148 if (USBUART_CD_currentTD.count != 0u)
149 {
150 requestHandled = USBUART_CD_InitControlRead();
151 }
152 }
153 else if (USBUART_CD_wValueHiReg == USBUART_CD_DESCR_HID_REPORT)
154 {
155 USBUART_CD_FindReportDescriptor();
156 if (USBUART_CD_currentTD.count != 0u)
157 {
158 requestHandled = USBUART_CD_InitControlRead();
159 }
160 }
161 else
162 {
163 /* Do not handle this request. */
164 }
165 break;
166
167 case USBUART_CD_HID_GET_REPORT:
168 USBUART_CD_FindReport();
169 if (USBUART_CD_currentTD.count != 0u)
170 {
171 requestHandled = USBUART_CD_InitControlRead();
172 }
173 break;
174
175 case USBUART_CD_HID_GET_IDLE:
176 /* This function does not support multiple reports per interface*/
177 /* Validate interfaceNumber and Report ID (should be 0): Do not support Idle per Report ID */
178 if ((interfaceNumber < USBUART_CD_MAX_INTERFACES_NUMBER) && (USBUART_CD_wValueLoReg == 0u))
179 {
180 USBUART_CD_currentTD.count = 1u;
181 USBUART_CD_currentTD.pData = &USBUART_CD_hidIdleRate[interfaceNumber];
182 requestHandled = USBUART_CD_InitControlRead();
183 }
184 break;
185
186 case USBUART_CD_HID_GET_PROTOCOL:
187 /* Validate interfaceNumber */
188 if( interfaceNumber < USBUART_CD_MAX_INTERFACES_NUMBER)
189 {
190 USBUART_CD_currentTD.count = 1u;
191 USBUART_CD_currentTD.pData = &USBUART_CD_hidProtocol[interfaceNumber];
192 requestHandled = USBUART_CD_InitControlRead();
193 }
194 break;
195
196 default: /* requestHandled is initialized as FALSE by default */
197 break;
198 }
199 }
200 else
201 {
202 /* Handle direction from host to device. */
203
204 switch (USBUART_CD_bRequestReg)
205 {
206 case USBUART_CD_HID_SET_REPORT:
207 USBUART_CD_FindReport();
208 if (USBUART_CD_currentTD.count != 0u)
209 {
210 requestHandled = USBUART_CD_InitControlWrite();
211 }
212 break;
213
214 case USBUART_CD_HID_SET_IDLE:
215 /* This function does not support multiple reports per interface */
216 /* Validate interfaceNumber and Report ID (should be 0): Do not support Idle per Report ID */
217 if ((interfaceNumber < USBUART_CD_MAX_INTERFACES_NUMBER) && (USBUART_CD_wValueLoReg == 0u))
218 {
219 USBUART_CD_hidIdleRate[interfaceNumber] = (uint8)USBUART_CD_wValueHiReg;
220 /* With regards to HID spec: "7.2.4 Set_Idle Request"
221 * Latency. If the current period has gone past the
222 * newly proscribed time duration, then a report
223 * will be generated immediately.
224 */
225 if(USBUART_CD_hidIdleRate[interfaceNumber] <
226 USBUART_CD_hidIdleTimer[interfaceNumber])
227 {
228 /* Set the timer to zero and let the UpdateHIDTimer() API return IDLE_TIMER_EXPIRED status*/
229 USBUART_CD_hidIdleTimer[interfaceNumber] = 0u;
230 }
231 /* If the new request is received within 4 milliseconds
232 * (1 count) of the end of the current period, then the
233 * new request will have no effect until after the report.
234 */
235 else if(USBUART_CD_hidIdleTimer[interfaceNumber] <= 1u)
236 {
237 /* Do nothing.
238 * Let the UpdateHIDTimer() API continue to work and
239 * return IDLE_TIMER_EXPIRED status
240 */
241 }
242 else
243 { /* Reload the timer*/
244 USBUART_CD_hidIdleTimer[interfaceNumber] =
245 USBUART_CD_hidIdleRate[interfaceNumber];
246 }
247 requestHandled = USBUART_CD_InitNoDataControlTransfer();
248 }
249 break;
250
251 case USBUART_CD_HID_SET_PROTOCOL:
252 /* Validate interfaceNumber and protocol (must be 0 or 1) */
253 if ((interfaceNumber < USBUART_CD_MAX_INTERFACES_NUMBER) && (USBUART_CD_wValueLoReg <= 1u))
254 {
255 USBUART_CD_hidProtocol[interfaceNumber] = (uint8)USBUART_CD_wValueLoReg;
256 requestHandled = USBUART_CD_InitNoDataControlTransfer();
257 }
258 break;
259
260 default:
261 /* Unknown class request is not handled. */
262 break;
263 }
264 }
265
266 return (requestHandled);
267}

286void USBUART_CD_FindHidClassDecriptor(void)
287{
288 const T_USBUART_CD_LUT CYCODE *pTmp;
289 volatile uint8 *pDescr;
290 uint8 interfaceN;
291
292 pTmp = USBUART_CD_GetConfigTablePtr(USBUART_CD_configuration - 1u);
293
294 interfaceN = (uint8) USBUART_CD_wIndexLoReg;
295 /* Third entry in the LUT starts the Interface Table pointers */
296 /* Now use the request interface number*/
297 pTmp = &pTmp[interfaceN + 2u];
298
299 /* USB_DEVICEx_CONFIGURATIONy_INTERFACEz_TABLE */
300 pTmp = (const T_USBUART_CD_LUT CYCODE *) pTmp->p_list;
301
302 /* Now use Alternate setting number */
303 pTmp = &pTmp[USBUART_CD_interfaceSetting[interfaceN]];
304
305 /* USB_DEVICEx_CONFIGURATIONy_INTERFACEz_ALTERNATEi_HID_TABLE */
306 pTmp = (const T_USBUART_CD_LUT CYCODE *) pTmp->p_list;
307
308 /* Fifth entry in the LUT points to Hid Class Descriptor in Configuration Descriptor */
309 pTmp = &pTmp[4u];
310 pDescr = (volatile uint8 *)pTmp->p_list;
311
312 /* The first byte contains the descriptor length */
313 USBUART_CD_currentTD.count = *pDescr;
314 USBUART_CD_currentTD.pData = pDescr;
315}

355

362 USBUART_CD_currentTD.pData = &pDescr[2u];
