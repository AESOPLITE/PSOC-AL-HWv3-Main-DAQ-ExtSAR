123void USBUART_CD_Suspend(void)
124{
125 uint8 enableInterrupts;
126
127 enableInterrupts = CyEnterCriticalSection();
128
129 if (0u != (USBUART_CD_CR0_REG & USBUART_CD_CR0_ENABLE))
130 {
131 /* USB block is enabled. */
132 USBUART_CD_backup.enableState = 1u;
133
134 #if (USBUART_CD_EP_MANAGEMENT_DMA)
135 USBUART_CD_Stop_DMA(USBUART_CD_MAX_EP);
136 #endif /* (USBUART_CD_EP_MANAGEMENT_DMA) */
137
138 #if (CY_PSOC4)
139 /* Suspend enter sequence. */
140 USBUART_CD_POWER_CTRL_REG |= (USBUART_CD_POWER_CTRL_SUSPEND |
141 USBUART_CD_POWER_CTRL_SUSPEND_DEL);
142
143 /* Store state of USB regulator and disable it. */
144 USBUART_CD_backup.mode = (uint8) (USBUART_CD_CR1_REG & USBUART_CD_CR1_REG_ENABLE);
145 USBUART_CD_CR1_REG &= (uint32) ~USBUART_CD_CR1_REG_ENABLE;
146
147 /* Store SIE interrupt sources. Valid bits are 0 - 4. */
148 USBUART_CD_backup.intrSeiMask = (uint8) USBUART_CD_INTR_SIE_MASK_REG;
149
150 #else
151 /* Ensure USB transmit enable is low (USB_USBIO_CR0.ten). - Manual Transmission - Disabled. */
152 USBUART_CD_USBIO_CR0_REG &= (uint8) ~USBUART_CD_USBIO_CR0_TEN;
153 CyDelayUs(USBUART_CD_WAIT_REG_STABILITY_50NS); /*~50ns delay. */
154
155 /* Disable the USBIO by asserting PM.USB_CR0.fsusbio_pd_n(Inverted) and pd_pullup_hv(Inverted) high. */
156 USBUART_CD_PM_USB_CR0_REG &= (uint8) ~(USBUART_CD_PM_USB_CR0_PD_N |
157 USBUART_CD_PM_USB_CR0_PD_PULLUP_N);
158
159 /* Disable the SIE. */
160 USBUART_CD_CR0_REG &= (uint8) ~USBUART_CD_CR0_ENABLE;
161
162 CyDelayUs(USBUART_CD_WAIT_REG_STABILITY_50NS); /* ~50ns delay. */
163 /* Store mode and disable VRegulator. */
164 USBUART_CD_backup.mode = (uint8) (USBUART_CD_CR1_REG & USBUART_CD_CR1_REG_ENABLE);
165 USBUART_CD_CR1_REG &= (uint8) ~USBUART_CD_CR1_REG_ENABLE;
166
167 CyDelayUs(USBUART_CD_WAIT_REG_STABILITY_1US); /* min 0.5us delay required. */
168
169 /* Disable the USBIO reference by setting PM.USB_CR0.fsusbio_ref_en.*/
170 USBUART_CD_PM_USB_CR0_REG &= (uint8) ~USBUART_CD_PM_USB_CR0_REF_EN;
171
172 /* Switch DP and DM terminals to GPIO mode and disconnect 1.5k pull-up. */
173 USBUART_CD_USBIO_CR1_REG |= USBUART_CD_USBIO_CR1_IOMODE;
174
175 /* Disable USBFS block. */
176 /* Clear power active and standby mode templates: disable USB block. */
177 USBUART_CD_PM_ACT_CFG_REG &= (uint8) ~USBUART_CD_PM_ACT_EN_FSUSB;
178 USBUART_CD_PM_STBY_CFG_REG &= (uint8) ~USBUART_CD_PM_STBY_EN_FSUSB;
179
180 CyDelayUs(USBUART_CD_WAIT_REG_STABILITY_1US); /* min 0.5us delay required. */
181 #endif /* (CY_PSOC4) */
182 }
183 else
184 {
185 USBUART_CD_backup.enableState = 0u;
186 }
187
188 CyExitCriticalSection(enableInterrupts);
189
190#if (USBUART_CD_DP_ISR_ACTIVE)
191 /* Clear active mode Dp interrupt source history. */
192 (void) USBUART_CD_Dp_ClearInterrupt();
193 CyIntClearPending(USBUART_CD_DP_INTC_VECT_NUM);
194
195 CyIntEnable (USBUART_CD_DP_INTC_VECT_NUM);
196#endif /* (USBUART_CD_DP_ISR_ACTIVE). */
197}

227void USBUART_CD_Resume(void)
228{
229 uint8 enableInterrupts;
230
231 enableInterrupts = CyEnterCriticalSection();
232
233 if (0u != USBUART_CD_backup.enableState)
234 {
235 #if (USBUART_CD_DP_ISR_ACTIVE)
236 CyIntDisable(USBUART_CD_DP_INTC_VECT_NUM);
237 #endif /* (USBUART_CD_DP_ISR_ACTIVE) */
238
239 #if (CY_PSOC4)
240 /* Enable clock to USB IP. */
241 USBUART_CD_USB_CLK_EN_REG |= USBUART_CD_USB_CLK_CSR_CLK_EN;
242
243 /* Restore arbiter configuration for DMA transfers. */
244 #if (USBUART_CD_EP_MANAGEMENT_DMA)
245 #if (USBUART_CD_ARB_ISR_ACTIVE)
246 /* Enable ARB EP interrupt sources. */
247 USBUART_CD_ARB_INT_EN_REG = USBUART_CD_DEFAULT_ARB_INT_EN;
248 #endif /* (USBUART_CD_EP_MANAGEMENT_DMA) */
249
250 /* Configure arbiter for Manual or Auto DMA operation and clear
251 * configuration completion.
252 */
253 USBUART_CD_ARB_CFG_REG = USBUART_CD_DEFAULT_ARB_CFG;
254 #endif /* (USBUART_CD_EP_MANAGEMENT_DMA) */
255
256 /* Restore level (hi, lo, med) for each interrupt source. */
257 USBUART_CD_INTR_LVL_SEL_REG = USBUART_CD_DEFAULT_INTR_LVL_SEL;
258
259 /* Store SIE interrupt sources. */
260 USBUART_CD_INTR_SIE_MASK_REG = (uint32) USBUART_CD_backup.intrSeiMask;
261
262 /* Set EP0.CR: ACK Setup, NAK IN/OUT. */
263 USBUART_CD_EP0_CR_REG = USBUART_CD_MODE_NAK_IN_OUT;
264
265 /* Restore data EP1-8 configuration. */
266 USBUART_CD_RestoreConfig();
267
268 /* Restore state of USB regulator and wait until it supples stable power. */
269 USBUART_CD_CR1_REG |= USBUART_CD_backup.mode;
270 CyDelayUs(USBUART_CD_WAIT_VREF_STABILITY);
271
272 /* Suspend exit sequence. */
273 USBUART_CD_POWER_CTRL_REG &= (uint32) ~USBUART_CD_POWER_CTRL_SUSPEND;
274 CyDelayUs(USBUART_CD_WAIT_SUSPEND_DEL_DISABLE);
275 USBUART_CD_POWER_CTRL_REG &= (uint32) ~USBUART_CD_POWER_CTRL_SUSPEND_DEL;
276
277 #else
278 /* Set power active and standby mode templates: enable USB block. */
279 USBUART_CD_PM_ACT_CFG_REG |= USBUART_CD_PM_ACT_EN_FSUSB;
280 USBUART_CD_PM_STBY_CFG_REG |= USBUART_CD_PM_STBY_EN_FSUSB;
281
282 /* Enable core clock. */
283 USBUART_CD_USB_CLK_EN_REG |= USBUART_CD_USB_CLK_ENABLE;
284
285 /* Enable the USBIO reference by setting PM.USB_CR0.fsusbio_ref_en.*/
286 USBUART_CD_PM_USB_CR0_REG |= USBUART_CD_PM_USB_CR0_REF_EN;
287
288 /* The reference is available ~40us after power restored. */
289 CyDelayUs(USBUART_CD_WAIT_VREF_RESTORE);
290 /* Restore state of USB regulator and wait until it supples stable power. */
291 USBUART_CD_CR1_REG |= USBUART_CD_backup.mode;
292 CyDelayUs(USBUART_CD_WAIT_VREF_STABILITY); /*~50ns delay. */
293
294 /* Enable USBIO. */
295 USBUART_CD_PM_USB_CR0_REG |= USBUART_CD_PM_USB_CR0_PD_N;
296 CyDelayUs(USBUART_CD_WAIT_PD_PULLUP_N_ENABLE);
297 /* Set the USBIO pull-up enable. */
298 USBUART_CD_PM_USB_CR0_REG |= USBUART_CD_PM_USB_CR0_PD_PULLUP_N;
299
300 /* Restore arbiter configuration for DMA transfers. */
301 #if (USBUART_CD_EP_MANAGEMENT_DMA)
302 #if (USBUART_CD_ARB_ISR_ACTIVE)
303 /* Enable ARB EP interrupt sources. */
304 USBUART_CD_ARB_INT_EN_REG = USBUART_CD_DEFAULT_ARB_INT_EN;
305 #endif /* (USBUART_CD_EP_MANAGEMENT_DMA) */
306
307 /* Configure arbiter for Manual or Auto DMA operation and clear
308 * configuration completion.
309 */
310 USBUART_CD_ARB_CFG_REG = USBUART_CD_DEFAULT_ARB_CFG;
311 #endif /* (USBUART_CD_EP_MANAGEMENT_DMA) */
312
313 /* Set EP0.CR: ACK Setup, STALL IN/OUT. */
314 USBUART_CD_EP0_CR_REG = USBUART_CD_MODE_STALL_IN_OUT;
315
316 /* Enable the USB IP to respond to USB traffic with the last address. */
317 USBUART_CD_CR0_REG |= USBUART_CD_CR0_ENABLE;
318 CyDelayCycles(USBUART_CD_WAIT_CR0_REG_STABILITY);
319
320 /* Enable D+ pull-up and keep USB control on IO. */
321 USBUART_CD_USBIO_CR1_REG = USBUART_CD_USBIO_CR1_USBPUEN;
322
323 /* Restore data EP1-8 configuration. */
324 USBUART_CD_RestoreConfig();
325 #endif /* (CY_PSOC4) */
326 }
327
328 CyExitCriticalSection(enableInterrupts);
329}
