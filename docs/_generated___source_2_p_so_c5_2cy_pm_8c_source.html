73void CyPmSaveClocks(void)
74{
75 /* Digital and analog clocks - save enable state and disable them all */
76 cyPmClockBackup.enClkA = CY_PM_ACT_CFG1_REG & CY_PM_ACT_EN_CLK_A_MASK;
77 cyPmClockBackup.enClkD = CY_PM_ACT_CFG2_REG;
78 CY_PM_ACT_CFG1_REG &= ((uint8)(~CY_PM_ACT_EN_CLK_A_MASK));
79 CY_PM_ACT_CFG2_REG &= ((uint8)(~CY_PM_ACT_EN_CLK_D_MASK));
80
81 /* Save current flash wait cycles and set the maximum value */
82 cyPmClockBackup.flashWaitCycles = CY_PM_CACHE_CR_CYCLES_MASK & CY_PM_CACHE_CR_REG;
83 CyFlash_SetWaitCycles(CY_PM_MAX_FLASH_WAIT_CYCLES);
84
85 /* IMO - save current IMO MHz OSC frequency and USB mode is on bit */
86 cyPmClockBackup.imoFreq = CY_PM_FASTCLK_IMO_CR_REG & CY_PM_FASTCLK_IMO_CR_FREQ_MASK;
87 cyPmClockBackup.imoUsbClk = CY_PM_FASTCLK_IMO_CR_REG & CY_PM_FASTCLK_IMO_CR_USB;
88
89 /* IMO doubler - save enable state */
90 if(0u != (CY_PM_FASTCLK_IMO_CR_REG & CY_PM_FASTCLK_IMO_CR_F2XON))
91 {
92 /* IMO doubler enabled - save and disable */
93 cyPmClockBackup.imo2x = CY_PM_ENABLED;
94 }
95 else
96 {
97 /* IMO doubler disabled */
98 cyPmClockBackup.imo2x = CY_PM_DISABLED;
99 }
100
101 /* Master clock - save source */
102 cyPmClockBackup.masterClkSrc = CY_PM_CLKDIST_MSTR1_REG & CY_PM_MASTER_CLK_SRC_MASK;
103
104 /* Switch Master clock's source from PLL's output to PLL's source */
105 if(CY_MASTER_SOURCE_PLL == cyPmClockBackup.masterClkSrc)
106 {
107 switch (CY_PM_CLKDIST_CR_REG & CY_PM_CLKDIST_PLL_SRC_MASK)
108 {
109 case CY_PM_CLKDIST_PLL_SRC_IMO:
110 CyMasterClk_SetSource(CY_MASTER_SOURCE_IMO);
111 break;
112
113 case CY_PM_CLKDIST_PLL_SRC_XTAL:
114 CyMasterClk_SetSource(CY_MASTER_SOURCE_XTAL);
115 break;
116
117 case CY_PM_CLKDIST_PLL_SRC_DSI:
118 CyMasterClk_SetSource(CY_MASTER_SOURCE_DSI);
119 break;
120
121 default:
122 CYASSERT(0u != 0u);
123 break;
124 }
125 }
126
127 /* PLL - check enable state, disable if needed */
128 if(0u != (CY_PM_FASTCLK_PLL_CFG0_REG & CY_PM_PLL_CFG0_ENABLE))
129 {
130 /* PLL is enabled - save state and disable */
131 cyPmClockBackup.pllEnableState = CY_PM_ENABLED;
132 CyPLL_OUT_Stop();
133 }
134 else
135 {
136 /* PLL is disabled - save state */
137 cyPmClockBackup.pllEnableState = CY_PM_DISABLED;
138 }
139
140 /* IMO - set appropriate frequency for LPM */
141 CyIMO_SetFreq(CY_PM_IMO_FREQ_LPM);
142
143 /* IMO - save enable state and enable without wait to settle */
144 if(0u != (CY_PM_ACT_CFG0_IMO & CY_PM_ACT_CFG0_REG))
145 {
146 /* IMO - save enabled state */
147 cyPmClockBackup.imoEnable = CY_PM_ENABLED;
148 }
149 else
150 {
151 /* IMO - save disabled state */
152 cyPmClockBackup.imoEnable = CY_PM_DISABLED;
153
154 /* Enable the IMO. Use software delay instead of the FTW-based inside */
155 CyIMO_Start(CY_PM_IMO_NO_WAIT_TO_SETTLE);
156
157 /* Settling time of the IMO is of the order of less than 6us */
158 CyDelayUs(6u);
159 }
160
161 /* IMO - save the current IMOCLK source and set to IMO if not yet */
162 if(0u != (CY_PM_FASTCLK_IMO_CR_REG & CY_PM_FASTCLK_IMO_CR_XCLKEN))
163 {
164 /* DSI or XTAL CLK */
165 cyPmClockBackup.imoClkSrc =
166 (0u == (CY_PM_CLKDIST_CR_REG & CY_PM_CLKDIST_IMO2X_SRC)) ? CY_IMO_SOURCE_DSI : CY_IMO_SOURCE_XTAL;
167
168 /* IMO - set IMOCLK source to IMO */
169 CyIMO_SetSource(CY_IMO_SOURCE_IMO);
170 }
171 else
172 {
173 /* IMO */
174 cyPmClockBackup.imoClkSrc = CY_IMO_SOURCE_IMO;
175 }
176
177 /* Save clk_imo source */
178 cyPmClockBackup.clkImoSrc = CY_PM_CLKDIST_CR_REG & CY_PM_CLKDIST_IMO_OUT_MASK;
179
180 /* If IMOCLK2X or SPC OSC is source for clk_imo, set it to IMOCLK */
181 if(CY_PM_CLKDIST_IMO_OUT_IMO != cyPmClockBackup.clkImoSrc)
182 {
183 /* Set IMOCLK to source for clk_imo */
184 CY_PM_CLKDIST_CR_REG = (CY_PM_CLKDIST_CR_REG & ((uint8)(~CY_PM_CLKDIST_IMO_OUT_MASK))) |
185 CY_PM_CLKDIST_IMO_OUT_IMO;
186 } /* Need to change nothing if IMOCLK is source clk_imo */
187
188 /* IMO doubler - disable it (saved above) */
189 if(0u != (CY_PM_FASTCLK_IMO_CR_REG & CY_PM_FASTCLK_IMO_CR_F2XON))
190 {
191 CyIMO_DisableDoubler();
192 }
193
194 /* Master clock - save divider and set it to divide-by-one (if no yet) */
195 cyPmClockBackup.clkSyncDiv = CY_PM_CLKDIST_MSTR0_REG;
196 if(CY_PM_DIV_BY_ONE != cyPmClockBackup.clkSyncDiv)
197 {
198 CyMasterClk_SetDivider(CY_PM_DIV_BY_ONE);
199 } /* No change if master clock divider is 1 */
200
201 /* Master clock source - set it to IMO if not yet. */
202 if(CY_MASTER_SOURCE_IMO != cyPmClockBackup.masterClkSrc)
203 {
204 CyMasterClk_SetSource(CY_MASTER_SOURCE_IMO);
205 } /* No change if master clock source is IMO */
206
207 /* Bus clock - save divider and set it, if needed, to divide-by-one */
208 cyPmClockBackup.clkBusDiv = (uint16) ((uint16) CY_PM_CLK_BUS_MSB_DIV_REG << 8u);
209 cyPmClockBackup.clkBusDiv |= CY_PM_CLK_BUS_LSB_DIV_REG;
210 if(CY_PM_BUS_CLK_DIV_BY_ONE != cyPmClockBackup.clkBusDiv)
211 {
212 CyBusClk_SetDivider(CY_PM_BUS_CLK_DIV_BY_ONE);
213 } /* Do nothing if saved and actual values are equal */
214
215 /* Set number of wait cycles for flash according to CPU frequency in MHz */
216 CyFlash_SetWaitCycles((uint8)CY_PM_GET_CPU_FREQ_MHZ);
217
218 /* MHz ECO - check enable state and disable if needed */
219 if(0u != (CY_PM_FASTCLK_XMHZ_CSR_REG & CY_PM_XMHZ_CSR_ENABLE))
220 {
221 /* MHz ECO is enabled - save state and disable */
222 cyPmClockBackup.xmhzEnableState = CY_PM_ENABLED;
223 CyXTAL_Stop();
224 }
225 else
226 {
227 /* MHz ECO is disabled - save state */
228 cyPmClockBackup.xmhzEnableState = CY_PM_DISABLED;
229 }
230
231
232
237 if(0u != (CY_PM_CLKDIST_DELAY_REG & CY_PM_CLKDIST_DELAY_EN))
238 {
239 cyPmClockBackup.clkDistDelay = CY_PM_ENABLED;
240 }
241 else
242 {
243 cyPmClockBackup.clkDistDelay = CY_PM_DISABLED;
244 }
245}

269{

271 uint16 i;
272 uint16 clkBusDivTmp;
273
274
275 /* Convertion table between CyIMO_SetFreq() parameters and register's value */
276 const uint8 CYCODE cyPmImoFreqMhz2Reg[7u] = {
277 CY_IMO_FREQ_12MHZ, CY_IMO_FREQ_6MHZ, CY_IMO_FREQ_24MHZ, CY_IMO_FREQ_3MHZ,
278 CY_IMO_FREQ_48MHZ, 5u, 6u};
279
280 /* Restore enable state of delay between system bus clock and ACLKs. */
281 if(CY_PM_ENABLED == cyPmClockBackup.clkDistDelay)
282 {
283 /* Delay for both bandgap and delay line to settle out */
284 CyDelayCycles((uint32)(CY_PM_CLK_DELAY_BANDGAP_SETTLE_US + CY_PM_CLK_DELAY_BIAS_SETTLE_US) *
285 CY_PM_GET_CPU_FREQ_MHZ);
286
287 CY_PM_CLKDIST_DELAY_REG |= CY_PM_CLKDIST_DELAY_EN;
288 }
289
290 /* MHz ECO restore state */
291 if(CY_PM_ENABLED == cyPmClockBackup.xmhzEnableState)
292 {
293
300 /* Enable XMHZ XTAL with no wait */
301 (void) CyXTAL_Start(CY_PM_XTAL_MHZ_NO_WAIT);
302
303 /* Read XERR bit to clear it */
304 (void) CY_PM_FASTCLK_XMHZ_CSR_REG;
305
306 /* Wait */
307 for(i = CY_PM_MHZ_XTAL_WAIT_NUM_OF_200_US; i > 0u; i--)
308 {
309 /* Make a 200 microseconds delay */
310 CyDelayCycles((uint32)CY_PM_WAIT_200_US * CY_PM_GET_CPU_FREQ_MHZ);
311
312 /* High output indicates oscillator failure */
313 if(0u == (CY_PM_FASTCLK_XMHZ_CSR_REG & CY_PM_XMHZ_CSR_XERR))
314 {
315 status = CYRET_SUCCESS;
316 break;
317 }
318 }
319
320 if(CYRET_TIMEOUT == status)
321 {
322
326 /* `#START_MHZ_ECO_TIMEOUT` */
327
328 /* `#END` */
329
330 #ifdef CY_BOOT_CY_PM_RESTORE_CLOCKS_ECO_TIMEOUT_CALLBACK
331 CyBoot_CyPmRestoreClocks_EcoTimeout_Callback();
332 #endif /* CY_BOOT_CY_PM_RESTORE_CLOCKS_ECO_TIMEOUT_CALLBACK */
333 }
334 } /* (CY_PM_ENABLED == cyPmClockBackup.xmhzEnableState) */
335
336
337 /* Temporary set maximum flash wait cycles */
338 CyFlash_SetWaitCycles(CY_PM_MAX_FLASH_WAIT_CYCLES);
339
340 /* XTAL and DSI clocks are ready to be source for Master clock. */
341 if((CY_PM_MASTER_CLK_SRC_XTAL == cyPmClockBackup.masterClkSrc) ||
342 (CY_PM_MASTER_CLK_SRC_DSI == cyPmClockBackup.masterClkSrc))
343 {
344 /* Restore Master clock's divider */
345 if(CY_PM_CLKDIST_MSTR0_REG != cyPmClockBackup.clkSyncDiv)
346 {
347 /* Restore Master clock divider */
348 CyMasterClk_SetDivider(cyPmClockBackup.clkSyncDiv);
349 }
350

352 CyMasterClk_SetSource(cyPmClockBackup.masterClkSrc);

357 (CY_IMO_FREQ_24MHZ == cyPmImoFreqMhz2Reg[cyPmClockBackup.imoFreq]))
358 {
359 /* Restore IMO frequency (24 MHz) and trim it for USB */
360 CyIMO_SetFreq(CY_IMO_FREQ_USB);
361 }
362 else
363 {
364 /* Restore IMO frequency */
365 CyIMO_SetFreq(cyPmImoFreqMhz2Reg[cyPmClockBackup.imoFreq]);
366
367 if(0u != (cyPmClockBackup.imoUsbClk & CY_PM_FASTCLK_IMO_CR_USB))
368 {
369 CY_PM_FASTCLK_IMO_CR_REG |= CY_PM_FASTCLK_IMO_CR_USB;
370 }
371 else
372 {
373 CY_PM_FASTCLK_IMO_CR_REG &= ((uint8)(~CY_PM_FASTCLK_IMO_CR_USB));
374 }
375 }
376
377 /* IMO - restore enable state if needed */
378 if((CY_PM_ENABLED == cyPmClockBackup.imoEnable) &&
379 (0u == (CY_PM_ACT_CFG0_IMO & CY_PM_ACT_CFG0_REG)))

383 }
384
385 /* IMO - restore IMOCLK source */
386 CyIMO_SetSource(cyPmClockBackup.imoClkSrc);
387
388 /* Restore IMO doubler enable state (turned off by CyPmSaveClocks()) */
389 if(CY_PM_ENABLED == cyPmClockBackup.imo2x)
390 {
391 CyIMO_EnableDoubler();
392 }
393
394 /* IMO - restore clk_imo source, if needed */
395 if(cyPmClockBackup.clkImoSrc != (CY_PM_CLKDIST_CR_REG & CY_PM_CLKDIST_IMO_OUT_MASK))
396 {
397 CY_PM_CLKDIST_CR_REG = (CY_PM_CLKDIST_CR_REG & ((uint8)(~CY_PM_CLKDIST_IMO_OUT_MASK))) |
398 cyPmClockBackup.clkImoSrc;
399 }

402 /* PLL restore state */
403 if(CY_PM_ENABLED == cyPmClockBackup.pllEnableState)
404 {
405
411 status = CYRET_TIMEOUT;
412
413 /* Enable PLL */
414 (void) CyPLL_OUT_Start(CY_PM_PLL_OUT_NO_WAIT);

416 /* Read to clear lock status after delay */
417 CyDelayUs((uint32)80u);
418 (void) CY_PM_FASTCLK_PLL_SR_REG;
419
420 /* It should take 250 us lock: 251-80 = 171 */
421 for(i = 171u; i > 0u; i--)

423 CyDelayUs((uint32)1u);
424
425 /* Accept PLL is OK after two consecutive polls indicate PLL lock */
426 if((0u != (CY_PM_FASTCLK_PLL_SR_REG & CY_PM_FASTCLK_PLL_LOCKED)) &&
427 (0u != (CY_PM_FASTCLK_PLL_SR_REG & CY_PM_FASTCLK_PLL_LOCKED)))

429 status = CYRET_SUCCESS;
430 break;
431 }
432 }
433
434 if(CYRET_TIMEOUT == status)
435 {
436
439 /* `#START_PLL_TIMEOUT` */
440
441 /* `#END` */
442
443 #ifdef CY_BOOT_CY_PM_RESTORE_CLOCKS_PLL_TIMEOUT_CALLBACK
444 CyBoot_CyPmRestoreClocks_PllTimeout_Callback();
445 #endif /* CY_BOOT_CY_PM_RESTORE_CLOCKS_PLL_TIMEOUT_CALLBACK */
446 }
447 } /* (CY_PM_ENABLED == cyPmClockBackup.pllEnableState) */
448
449
450 /* PLL and IMO is ready to be source for Master clock */
451 if((CY_PM_MASTER_CLK_SRC_IMO == cyPmClockBackup.masterClkSrc) ||
452 (CY_PM_MASTER_CLK_SRC_PLL == cyPmClockBackup.masterClkSrc))
453 {
454 /* Restore Master clock divider */
455 if(CY_PM_CLKDIST_MSTR0_REG != cyPmClockBackup.clkSyncDiv)
456 {
457 CyMasterClk_SetDivider(cyPmClockBackup.clkSyncDiv);
458 }
459
460 /* Restore Master clock source */
461 CyMasterClk_SetSource(cyPmClockBackup.masterClkSrc);
462 }
463
464 /* IMO - disable if it was originally disabled */
465 if((CY_PM_DISABLED == cyPmClockBackup.imoEnable) &&
466 (0u != (CY_PM_ACT_CFG0_IMO & CY_PM_ACT_CFG0_REG)))
467 {
468 CyIMO_Stop();
469 }
470
471 /* Bus clock - restore divider, if needed */
472 clkBusDivTmp = (uint16) ((uint16)CY_PM_CLK_BUS_MSB_DIV_REG << 8u);
473 clkBusDivTmp |= CY_PM_CLK_BUS_LSB_DIV_REG;
474 if(cyPmClockBackup.clkBusDiv != clkBusDivTmp)
475 {
476 CyBusClk_SetDivider(cyPmClockBackup.clkBusDiv);
477 }
478
479 /* Restore flash wait cycles */
480 CY_PM_CACHE_CR_REG = ((CY_PM_CACHE_CR_REG & ((uint8)(~CY_PM_CACHE_CR_CYCLES_MASK))) |
481 cyPmClockBackup.flashWaitCycles);
482
483 /* Digital and analog clocks - restore state */
484 CY_PM_ACT_CFG1_REG = cyPmClockBackup.enClkA;
485 CY_PM_ACT_CFG2_REG = cyPmClockBackup.enClkD;
486}
