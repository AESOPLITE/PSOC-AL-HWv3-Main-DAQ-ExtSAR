494void SPIM_BP_WriteTxData(uint8 txData)
495{
496 #if(SPIM_BP_TX_SOFTWARE_BUF_ENABLED)
497
498 uint8 tempStatus;
499 uint8 tmpTxBufferRead;
500
501 /* Block if TX buffer is FULL: don't overwrite */
502 do
503 {
504 tmpTxBufferRead = SPIM_BP_txBufferRead;
505 if(0u == tmpTxBufferRead)
506 {
507 tmpTxBufferRead = (SPIM_BP_TX_BUFFER_SIZE - 1u);
508 }
509 else
510 {
511 tmpTxBufferRead--;
512 }
513
514 }while(tmpTxBufferRead == SPIM_BP_txBufferWrite);
515
516 /* Disable TX interrupt to protect global veriables */
517 SPIM_BP_DisableTxInt();
518
519 tempStatus = SPIM_BP_GET_STATUS_TX(SPIM_BP_swStatusTx);
520 SPIM_BP_swStatusTx = tempStatus;
521
522
523 if((SPIM_BP_txBufferRead == SPIM_BP_txBufferWrite) &&
524 (0u != (SPIM_BP_swStatusTx & SPIM_BP_STS_TX_FIFO_NOT_FULL)))
525 {
526 /* Put data element into the TX FIFO */
527 CY_SET_REG8(SPIM_BP_TXDATA_PTR, txData);
528 }
529 else
530 {
531 /* Add to the TX software buffer */
532 SPIM_BP_txBufferWrite++;
533 if(SPIM_BP_txBufferWrite >= SPIM_BP_TX_BUFFER_SIZE)
534 {
535 SPIM_BP_txBufferWrite = 0u;
536 }
537
538 if(SPIM_BP_txBufferWrite == SPIM_BP_txBufferRead)
539 {
540 SPIM_BP_txBufferRead++;
541 if(SPIM_BP_txBufferRead >= SPIM_BP_TX_BUFFER_SIZE)
542 {
543 SPIM_BP_txBufferRead = 0u;
544 }
545 SPIM_BP_txBufferFull = 1u;
546 }
547
548 SPIM_BP_txBuffer[SPIM_BP_txBufferWrite] = txData;
549
550 SPIM_BP_TX_STATUS_MASK_REG |= SPIM_BP_STS_TX_FIFO_NOT_FULL;
551 }
552
553 SPIM_BP_EnableTxInt();
554
555 #else
556 /* Wait until TX FIFO has a place */
557 while(0u == (SPIM_BP_TX_STATUS_REG & SPIM_BP_STS_TX_FIFO_NOT_FULL))
558 {
559 }
560
561 /* Put data element into the TX FIFO */
562 CY_SET_REG8(SPIM_BP_TXDATA_PTR, txData);
563
564 #endif /* (SPIM_BP_TX_SOFTWARE_BUF_ENABLED) */
