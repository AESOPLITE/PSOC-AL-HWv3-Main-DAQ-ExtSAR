80uint8 I2C_RTC_MasterWriteBuf(uint8 slaveAddress, uint8 * wrData, uint8 cnt, uint8 mode)
81
82{
83 uint8 errStatus;
84
85 errStatus = I2C_RTC_MSTR_NOT_READY;
86
87 if(NULL != wrData)
88 {
89 /* Check I2C state to allow transfer: valid states are IDLE or HALT */
90 if(I2C_RTC_SM_IDLE == I2C_RTC_state)
91 {
92 /* Master is ready for transaction: check if bus is free */
93 if(I2C_RTC_CHECK_BUS_FREE(I2C_RTC_MCSR_REG))
94 {
95 errStatus = I2C_RTC_MSTR_NO_ERROR;
96 }
97 else
98 {
99 errStatus = I2C_RTC_MSTR_BUS_BUSY;
100 }
101 }
102 else if(I2C_RTC_SM_MSTR_HALT == I2C_RTC_state)
103 {
104 /* Master is ready and waiting for ReStart */
105 errStatus = I2C_RTC_MSTR_NO_ERROR;
106
107 I2C_RTC_ClearPendingInt();
108 I2C_RTC_mstrStatus &= (uint8) ~I2C_RTC_MSTAT_XFER_HALT;
109 }
110 else
111 {
112 /* errStatus = I2C_RTC_MSTR_NOT_READY was send before */
113 }
114
115 if(I2C_RTC_MSTR_NO_ERROR == errStatus)
116 {
117 /* Set state to start write transaction */
118 I2C_RTC_state = I2C_RTC_SM_MSTR_WR_ADDR;
119
120 /* Prepare write buffer */
121 I2C_RTC_mstrWrBufIndex = 0u;
122 I2C_RTC_mstrWrBufSize = cnt;
123 I2C_RTC_mstrWrBufPtr = (volatile uint8 *) wrData;
124
125 /* Set end of transaction flag: Stop or Halt (following ReStart) */
126 I2C_RTC_mstrControl = mode;
127
128 /* Clear write status history */
129 I2C_RTC_mstrStatus &= (uint8) ~I2C_RTC_MSTAT_WR_CMPLT;
130
131 /* Hardware actions: write address and generate Start or ReStart */
132 I2C_RTC_DATA_REG = (uint8) (slaveAddress << I2C_RTC_SLAVE_ADDR_SHIFT);
133
134 if(I2C_RTC_CHECK_RESTART(mode))
135 {
136 I2C_RTC_GENERATE_RESTART;
137 }
138 else
139 {
140 I2C_RTC_GENERATE_START;
141 }
142
143 /* Enable interrupt to complete transfer */
144 I2C_RTC_EnableInt();
145 }
146 }
147
148 return(errStatus);
149}

295uint8 I2C_RTC_MasterSendStart(uint8 slaveAddress, uint8 R_nW)
296
297{
298 uint8 errStatus;
299
300 errStatus = I2C_RTC_MSTR_NOT_READY;
301
302 /* If IDLE, check if bus is free */
303 if(I2C_RTC_SM_IDLE == I2C_RTC_state)
304 {
305 /* If bus is free, generate Start condition */
306 if(I2C_RTC_CHECK_BUS_FREE(I2C_RTC_MCSR_REG))
307 {
308 /* Disable interrupt for manual master operation */
309 I2C_RTC_DisableInt();
310
311 /* Set address and read/write flag */
312 slaveAddress = (uint8) (slaveAddress << I2C_RTC_SLAVE_ADDR_SHIFT);
313 if(0u != R_nW)
314 {
315 slaveAddress |= I2C_RTC_READ_FLAG;
316 I2C_RTC_state = I2C_RTC_SM_MSTR_RD_ADDR;
317 }
318 else
319 {
320 I2C_RTC_state = I2C_RTC_SM_MSTR_WR_ADDR;
321 }
322
323 /* Hardware actions: write address and generate Start */
324 I2C_RTC_DATA_REG = slaveAddress;
325 I2C_RTC_GENERATE_START_MANUAL;
326
327 /* Wait until address is transferred */
328 while(I2C_RTC_WAIT_BYTE_COMPLETE(I2C_RTC_CSR_REG))
329 {
330 }
331
332 #if(I2C_RTC_MODE_MULTI_MASTER_SLAVE_ENABLED)
333 if(I2C_RTC_CHECK_START_GEN(I2C_RTC_MCSR_REG))
334 {
335 I2C_RTC_CLEAR_START_GEN;
336
337 /* Start condition was not generated: reset FSM to IDLE */
338 I2C_RTC_state = I2C_RTC_SM_IDLE;
339 errStatus = I2C_RTC_MSTR_ERR_ABORT_START_GEN;
340 }
341 else
342 #endif /* (I2C_RTC_MODE_MULTI_MASTER_SLAVE_ENABLED) */
343
344 #if(I2C_RTC_MODE_MULTI_MASTER_ENABLED)
345 if(I2C_RTC_CHECK_LOST_ARB(I2C_RTC_CSR_REG))
346 {
347 I2C_RTC_BUS_RELEASE_MANUAL;
348
349 /* Master lost arbitrage: reset FSM to IDLE */
350 I2C_RTC_state = I2C_RTC_SM_IDLE;
351 errStatus = I2C_RTC_MSTR_ERR_ARB_LOST;
352 }
353 else
354 #endif /* (I2C_RTC_MODE_MULTI_MASTER_ENABLED) */
355
356 if(I2C_RTC_CHECK_ADDR_NAK(I2C_RTC_CSR_REG))
357 {
358 /* Address has been NACKed: reset FSM to IDLE */
359 I2C_RTC_state = I2C_RTC_SM_IDLE;
360 errStatus = I2C_RTC_MSTR_ERR_LB_NAK;
361 }
362 else
363 {
364 /* Start was sent without errors */
365 errStatus = I2C_RTC_MSTR_NO_ERROR;
366 }
367 }
368 else
369 {
370 errStatus = I2C_RTC_MSTR_BUS_BUSY;
371 }
372 }
373
374 return(errStatus);
375}

623uint8 I2C_RTC_MasterReadByte(uint8 acknNak)
624{
625 uint8 theByte;
626
627 theByte = 0u;
628
629 /* Check if START condition was generated */
630 if(I2C_RTC_CHECK_MASTER_MODE(I2C_RTC_MCSR_REG))
631 {
632 /* When address phase needs to release bus and receive byte,
633 * then decide ACK or NACK
634 */
635 if(I2C_RTC_SM_MSTR_RD_ADDR == I2C_RTC_state)
636 {
637 I2C_RTC_READY_TO_READ_MANUAL;
638 I2C_RTC_state = I2C_RTC_SM_MSTR_RD_DATA;
639 }
640
641 /* Wait until data byte has been received */
642 while(I2C_RTC_WAIT_BYTE_COMPLETE(I2C_RTC_CSR_REG))
643 {
644 }
645
646 theByte = I2C_RTC_DATA_REG;
647
648 /* Command ACK to receive next byte and continue transfer.
649 * Do nothing for NACK. The NACK will be generated by
650 * Stop or ReStart routine.
651 */
652 if(acknNak != 0u) /* Generate ACK */
653 {
654 I2C_RTC_ACK_AND_RECEIVE_MANUAL;
655 }
656 else /* Do nothing for the follwong NACK */
657 {
658 I2C_RTC_state = I2C_RTC_SM_MSTR_HALT;
659 }
660 }
661
662 return(theByte);
663}
