425void USBUART_CD_PutString(const char8 string[])
426{
427 uint16 strLength;
428 uint16 sendLength;
429 uint16 bufIndex = 0u;
430
431 uint8 epNumber = USBUART_CD_cdcDataInEp[USBUART_CD_activeCom];
432
433 /* Get length string length (it is terminated with zero). */
434 strLength = USBUART_CD_StrLen(string);
435
436 do
437 {
438 /* Limit length to maximum packet size of endpoint. */
439 sendLength = (strLength > USBUART_CD_EP[epNumber].bufferSize) ?
440 USBUART_CD_EP[epNumber].bufferSize : strLength;
441
442 /* Load IN endpoint and expose it to host. */
443 USBUART_CD_LoadInEP(epNumber, (const uint8 *)&string[bufIndex], sendLength);
444 strLength -= sendLength;
445
446 /* If more data are present to send or full packet was sent */
447 if ((strLength > 0u) || (sendLength == USBUART_CD_EP[epNumber].bufferSize))
448 {
449 bufIndex += sendLength;
450
451 /* Wait until host read data from IN endpoint buffer. */
452 while (USBUART_CD_IN_BUFFER_FULL == USBUART_CD_EP[epNumber].apiEpState)
453 {
454 }
455
456 /* If last packet is exactly maximum packet size, it shall be followed
457 * by a zero-length packet to assure the end of segment is properly
458 * identified by the terminal.
459 */
460 if (0u == strLength)
461 {
462 USBUART_CD_LoadInEP(epNumber, NULL, 0u);
463 }
464 }
465 }
466 while (strLength > 0u);
467}
