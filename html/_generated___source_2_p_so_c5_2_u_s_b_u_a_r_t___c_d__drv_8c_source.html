94CY_ISR(USBUART_CD_EP_0_ISR)
95{
96 uint8 tempReg;
97 uint8 modifyReg;
98
99#ifdef USBUART_CD_EP_0_ISR_ENTRY_CALLBACK
100 USBUART_CD_EP_0_ISR_EntryCallback();
101#endif /* (USBUART_CD_EP_0_ISR_ENTRY_CALLBACK) */
102
103 tempReg = USBUART_CD_EP0_CR_REG;
104 if ((tempReg & USBUART_CD_MODE_ACKD) != 0u)
105 {
106 modifyReg = 1u;
107 if ((tempReg & USBUART_CD_MODE_SETUP_RCVD) != 0u)
108 {
109 if ((tempReg & USBUART_CD_MODE_MASK) != USBUART_CD_MODE_NAK_IN_OUT)
110 {
111 /* Mode not equal to NAK_IN_OUT: invalid setup */
112 modifyReg = 0u;
113 }
114 else
115 {
116 USBUART_CD_HandleSetup();
117
118 if ((USBUART_CD_ep0Mode & USBUART_CD_MODE_SETUP_RCVD) != 0u)
119 {
120 /* SETUP bit set: exit without mode modificaiton */
121 modifyReg = 0u;
122 }
123 }
124 }
125 else if ((tempReg & USBUART_CD_MODE_IN_RCVD) != 0u)
126 {
127 USBUART_CD_HandleIN();
128 }
129 else if ((tempReg & USBUART_CD_MODE_OUT_RCVD) != 0u)
130 {
131 USBUART_CD_HandleOUT();
132 }
133 else
134 {
135 modifyReg = 0u;
136 }
137
138 /* Modify the EP0_CR register */
139 if (modifyReg != 0u)
140 {
141
142 tempReg = USBUART_CD_EP0_CR_REG;
143
144 /* Make sure that SETUP bit is cleared before modification */
145 if ((tempReg & USBUART_CD_MODE_SETUP_RCVD) == 0u)
146 {
147 /* Update count register */
148 tempReg = (uint8) USBUART_CD_ep0Toggle | USBUART_CD_ep0Count;
149 USBUART_CD_EP0_CNT_REG = tempReg;
150
151 /* Make sure that previous write operaiton was successful */
152 if (tempReg == USBUART_CD_EP0_CNT_REG)
153 {
154 /* Repeat until next successful write operation */
155 do
156 {
157 /* Init temporary variable */
158 modifyReg = USBUART_CD_ep0Mode;
159
160 /* Unlock register */
161 tempReg = (uint8) (USBUART_CD_EP0_CR_REG & USBUART_CD_MODE_SETUP_RCVD);
162
163 /* Check if SETUP bit is not set */
164 if (0u == tempReg)
165 {
166 /* Set the Mode Register */
167 USBUART_CD_EP0_CR_REG = USBUART_CD_ep0Mode;
168
169 /* Writing check */
170 modifyReg = USBUART_CD_EP0_CR_REG & USBUART_CD_MODE_MASK;
171 }
172 }
173 while (modifyReg != USBUART_CD_ep0Mode);
174 }
175 }
176 }

198void USBUART_CD_HandleSetup(void)
199{
200 uint8 requestHandled;
201
202 /* Clear register lock by SIE (read register) and clear setup bit
203 * (write any value in register).
204 */
205 requestHandled = (uint8) USBUART_CD_EP0_CR_REG;
206 USBUART_CD_EP0_CR_REG = (uint8) requestHandled;
207 requestHandled = (uint8) USBUART_CD_EP0_CR_REG;
208
209 if ((requestHandled & USBUART_CD_MODE_SETUP_RCVD) != 0u)
210 {
211 /* SETUP bit is set: exit without mode modification. */
212 USBUART_CD_ep0Mode = requestHandled;
213 }
214 else
215 {
216 /* In case the previous transfer did not complete, close it out */
217 USBUART_CD_UpdateStatusBlock(USBUART_CD_XFER_PREMATURE);
218
219 /* Check request type. */
220 switch (USBUART_CD_bmRequestTypeReg & USBUART_CD_RQST_TYPE_MASK)
221 {
222 case USBUART_CD_RQST_TYPE_STD:
223 requestHandled = USBUART_CD_HandleStandardRqst();
224 break;
225
226 case USBUART_CD_RQST_TYPE_CLS:
227 requestHandled = USBUART_CD_DispatchClassRqst();
228 break;
229
230 case USBUART_CD_RQST_TYPE_VND:
231 requestHandled = USBUART_CD_HandleVendorRqst();
232 break;
233
234 default:
235 requestHandled = USBUART_CD_FALSE;
236 break;
237 }
238
239 /* If request is not recognized. Stall endpoint 0 IN and OUT. */
240 if (requestHandled == USBUART_CD_FALSE)
241 {
242 USBUART_CD_ep0Mode = USBUART_CD_MODE_STALL_IN_OUT;
243 }
244 }
245}

357void USBUART_CD_LoadEP0(void)
358{
359 uint8 ep0Count = 0u;
360
361 /* Update the transfer byte count from the last transaction */
362 USBUART_CD_transferByteCount += USBUART_CD_lastPacketSize;
363
364 /* Now load the next transaction */
365 while ((USBUART_CD_currentTD.count > 0u) && (ep0Count < 8u))
366 {
367 USBUART_CD_EP0_DR_BASE.epData[ep0Count] = (uint8) *USBUART_CD_currentTD.pData;
368 USBUART_CD_currentTD.pData = &USBUART_CD_currentTD.pData[1u];
369 ep0Count++;
370 USBUART_CD_currentTD.count--;
371 }
372
373 /* Support zero-length packet */
374 if ((USBUART_CD_lastPacketSize == 8u) || (ep0Count > 0u))
375 {
376 /* Update the data toggle */
377 USBUART_CD_ep0Toggle ^= USBUART_CD_EP0_CNT_DATA_TOGGLE;
378 /* Set the Mode Register */
379 USBUART_CD_ep0Mode = USBUART_CD_MODE_ACK_IN_STATUS_OUT;
380 /* Update the state (or stay the same) */
381 USBUART_CD_transferState = USBUART_CD_TRANS_STATE_CONTROL_READ;
382 }
383 else
384 {
385 /* Expect Status Stage Out */
386 USBUART_CD_ep0Mode = USBUART_CD_MODE_STATUS_OUT_ONLY;
387 /* Update the state (or stay the same) */
388 USBUART_CD_transferState = USBUART_CD_TRANS_STATE_CONTROL_READ;
389 }
390
391 /* Save the packet size for next time */
392 USBUART_CD_ep0Count = (uint8) ep0Count;
393 USBUART_CD_lastPacketSize = (uint8) ep0Count;
394}

611void USBUART_CD_ControlWriteDataStage(void)
612{
613 uint8 ep0Count;
614 uint8 regIndex = 0u;
615
616 ep0Count = (USBUART_CD_EP0_CNT_REG & USBUART_CD_EPX_CNT0_MASK) - USBUART_CD_EPX_CNTX_CRC_COUNT;
617
618 USBUART_CD_transferByteCount += (uint8)ep0Count;
619
620 while ((USBUART_CD_currentTD.count > 0u) && (ep0Count > 0u))
621 {
622 *USBUART_CD_currentTD.pData = (uint8) USBUART_CD_EP0_DR_BASE.epData[regIndex];
623 USBUART_CD_currentTD.pData = &USBUART_CD_currentTD.pData[1u];
624 regIndex++;
625 ep0Count--;
626 USBUART_CD_currentTD.count--;
627 }
628
629 USBUART_CD_ep0Count = (uint8)ep0Count;
630
631 /* Update the data toggle */
632 USBUART_CD_ep0Toggle ^= USBUART_CD_EP0_CNT_DATA_TOGGLE;
633
634 /* Expect Data or Status Stage */
635 USBUART_CD_ep0Mode = USBUART_CD_MODE_ACK_OUT_STATUS_IN;
