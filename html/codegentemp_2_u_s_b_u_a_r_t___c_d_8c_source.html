216
217 /* The internal regulator (CR1.REG_ENABLE) is enabled in
218 * USBUART_CD_InitComponent() if it is required.
219 */
220
221 /* Enable USBIO control on drive mode of D+ and D- pins. */
222 USBUART_CD_USBIO_CR1_REG &= ~ (uint32) USBUART_CD_USBIO_CR1_IOMODE;
223
224 /* Set number of LF CLK to detect UBS bus reset. */
225 USBUART_CD_BUS_RST_CNT_REG = USBUART_CD_DEFUALT_BUS_RST_CNT;
226
227 /* Select VBUS detection source and clear PHY isolate. The application level
228 * must ensure that VBUS is valid. There is no need to wait 2us before VBUS is valid.
229 */
230 USBUART_CD_POWER_CTRL_REG = USBUART_CD_DEFAULT_POWER_CTRL_VBUS;
231
232 /* Enable PHY detector and single-ended and differential receivers. */
233 USBUART_CD_POWER_CTRL_REG |= USBUART_CD_DEFAULT_POWER_CTRL_PHY;
234
235 /* Suspend clear sequence. */
236 USBUART_CD_POWER_CTRL_REG &= (uint32) ~USBUART_CD_POWER_CTRL_SUSPEND;
237 CyDelayUs(USBUART_CD_WAIT_SUSPEND_DEL_DISABLE);
238 USBUART_CD_POWER_CTRL_REG &= (uint32) ~USBUART_CD_POWER_CTRL_SUSPEND_DEL;
239
240 /* Sets IMO lock options and clear all other bits. */
241 USBUART_CD_CR1_REG = USBUART_CD_DEFUALT_CR1;
242
243 /* Configure level (hi, lo, med) for each interrupt source. */
244 USBUART_CD_INTR_LVL_SEL_REG = USBUART_CD_DEFAULT_INTR_LVL_SEL;
245
246 /* Configure interrupt sources from: SOF, Bus Reset and EP0. */
247 USBUART_CD_INTR_SIE_MASK_REG = USBUART_CD_DEFAULT_INTR_SIE_MASK;
248
249#else
250 uint8 enableInterrupts = CyEnterCriticalSection();

254#endif /* (USBUART_CD_EP_MANAGEMENT_DMA) */

266 /* ENABLING USBIO PADS IN USB MODE FROM I/O MODE. */
267 /* Ensure USB transmit enable is low (USB_USBIO_CR0.ten). - Manual Transmission - Disabled. */

269 CyDelayUs(USBUART_CD_WAIT_REG_STABILITY_50NS); /* ~50ns delay. */
270 /* Disable USBIO by asserting PM.USB_CR0.fsusbio_pd_n(Inverted.
271 * high. These bits will be set low by the power manager out-of-reset.
272 * Also confirm USBIO pull-up is disabled.

274 USBUART_CD_PM_USB_CR0_REG &= (uint8) ~(USBUART_CD_PM_USB_CR0_PD_N |
275 USBUART_CD_PM_USB_CR0_PD_PULLUP_N);

281 USBUART_CD_PM_USB_CR0_REG |= USBUART_CD_PM_USB_CR0_REF_EN;
282 /* Reference is available for 1us after regulator is enabled. */
283 CyDelayUs(USBUART_CD_WAIT_REG_STABILITY_1US);
284 /* OR 40us after power is restored. */
285 CyDelayUs(USBUART_CD_WAIT_VREF_RESTORE);
286 /* Ensure single-ended disable bits are low (PRT15.INP_DIS[7:6])(input receiver enabled). */
287 USBUART_CD_DM_INP_DIS_REG &= (uint8) ~USBUART_CD_DM_MASK;
288 USBUART_CD_DP_INP_DIS_REG &= (uint8) ~USBUART_CD_DP_MASK;
289
290 /* Enable USBIO. */
291 USBUART_CD_PM_USB_CR0_REG |= USBUART_CD_PM_USB_CR0_PD_N;
292 CyDelayUs(USBUART_CD_WAIT_PD_PULLUP_N_ENABLE);
293 /* Set USBIO pull-up enable. */
294 USBUART_CD_PM_USB_CR0_REG |= USBUART_CD_PM_USB_CR0_PD_PULLUP_N;
295
296 /* Reset Arbiter Write Address register for endpoint 1. */
297 CY_SET_REG8(USBUART_CD_ARB_RW1_WA_PTR, 0u);
298 CY_SET_REG8(USBUART_CD_ARB_RW1_WA_MSB_PTR, 0u);
299
300#if (USBUART_CD_EP_MANAGEMENT_DMA)

306 #if (USBUART_CD_EP_MANAGEMENT_DMA_AUTO && (USBUART_CD_EP_DMA_AUTO_OPT == 0u))

308 #endif /* (USBUART_CD_EP_MANAGEMENT_DMA_AUTO && (USBUART_CD_EP_DMA_AUTO_OPT == 0u)) */
309 }
310#endif /* (USBUART_CD_EP_MANAGEMENT_DMA) */

312 CyExitCriticalSection(enableInterrupts);
313#endif /* (CY_PSOC4) */
314
315 /* Configure interrupts from USB block. */
316#if (CY_PSOC4)
317 /* Configure hi_int: set handler and priority. */

320
321 /* Configure lo_int: set handler and priority. */
322 CyIntSetPriority (USBUART_CD_INTR_LO_VECT_NUM, USBUART_CD_INTR_LO_PRIORITY);
323 (void) CyIntSetVector(USBUART_CD_INTR_LO_VECT_NUM, &USBUART_CD_INTR_LO_ISR);
324
325 /* Configure med_int: set handler and priority (routed through DSI). */
326 CyIntSetPriority (USBUART_CD_INTR_MED_VECT_NUM, USBUART_CD_INTR_MED_PRIORITY);
327 (void) CyIntSetVector(USBUART_CD_INTR_MED_VECT_NUM, &USBUART_CD_INTR_MED_ISR);
328
329#else
330 /* Set bus reset interrupt. */
331 CyIntSetPriority(USBUART_CD_BUS_RESET_VECT_NUM, USBUART_CD_BUS_RESET_PRIOR);
332 (void) CyIntSetVector(USBUART_CD_BUS_RESET_VECT_NUM, &USBUART_CD_BUS_RESET_ISR);

531#endif /* (CY_PSOC4) */

570#endif /* (USBUART_CD_EP_MANAGEMENT_DMA) */

584

586 #if (USBUART_CD_VDDD_MV < USBUART_CD_3500MV)

592 #endif /* (USBUART_CD_VDDD_MV < USBUART_CD_3500MV) */
593 break;

606 #if (USBUART_CD_LPM_ACTIVE)
607 if (NULL != USBUART_CD_GetBOSPtr())
608 {

610 * Reset NYET_EN and SUB_RESP bits in the LPM_CTRL register.
611 */

613 USBUART_CD_LPM_CTRL_LPM_ACK_RESP);
614 }
615 else
616 {
617 /* Disable LPM for active device. */
618 USBUART_CD_LPM_CTRL_REG &= (uint32) ~USBUART_CD_LPM_CTRL_LPM_EN;
619 }
620 #endif /* (USBUART_CD_LPM_ACTIVE) */
621
622 /* Enable device to responds to USB traffic with address 0. */
623 USBUART_CD_CR0_REG = USBUART_CD_DEFUALT_CR0;
624
625#else
626 /* Set EP0.CR: ACK Setup, STALL IN/OUT. */
627 USBUART_CD_EP0_CR_REG = USBUART_CD_MODE_STALL_IN_OUT;
628

1342void USBUART_CD_LoadInEP(uint8 epNumber, const uint8 pData[], uint16 length)
1343
1344{
1345 if ((epNumber > USBUART_CD_EP0) && (epNumber < USBUART_CD_MAX_EP))
1346 {
1347 #if (!USBUART_CD_EP_MANAGEMENT_DMA_AUTO)
1348 /* Limit length to available buffer USB IP buffer size.*/
1349 if (length > (USBUART_CD_EPX_DATA_BUF_MAX - USBUART_CD_EP[epNumber].buffOffset))
1350 {
1351 length = USBUART_CD_EPX_DATA_BUF_MAX - USBUART_CD_EP[epNumber].buffOffset;
1352 }
1353 #endif /* (!USBUART_CD_EP_MANAGEMENT_DMA_AUTO) */
1354
1355 /* Set count and data toggle. */
1356 USBUART_CD_SIE_EP_BASE.sieEp[epNumber].epCnt0 = (uint8) HI8(length) | USBUART_CD_EP[epNumber].epToggle;
1357 USBUART_CD_SIE_EP_BASE.sieEp[epNumber].epCnt1 = (uint8) LO8(length);
1358
1359 #if (USBUART_CD_EP_MANAGEMENT_MANUAL)
1360 if (NULL != pData)
1361 {
1362 /* Copy data using arbiter data register. */
1363 uint16 i;
1364 for (i = 0u; i < length; ++i)
1365 {
1366 USBUART_CD_ARB_EP_BASE.arbEp[epNumber].rwDr = pData[i];
1367 }
1368 }
1369
1370 /* IN endpoint buffer is full - read to be read. */
1371 USBUART_CD_EP[epNumber].apiEpState = USBUART_CD_NO_EVENT_PENDING;
1372
1373 /* Arm IN endpoint. */
1374 USBUART_CD_SIE_EP_BASE.sieEp[epNumber].epCr0 = USBUART_CD_EP[epNumber].epMode;
1375
1376 #else
1377
1378 #if (CY_PSOC3 || CY_PSOC5LP)
1379 /* Initialize DMA if it was not initialized. */
1380 if (DMA_INVALID_TD == USBUART_CD_DmaTd[epNumber])
1381 {
1382 USBUART_CD_InitEP_DMA(epNumber, pData);
1383 }
