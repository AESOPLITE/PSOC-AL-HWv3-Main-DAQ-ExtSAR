43void Timer_SelLow_Init(void)
44{
45 #if(!Timer_SelLow_UsingFixedFunction)
46 /* Interrupt State Backup for Critical Region*/
47 uint8 Timer_SelLow_interruptState;
48 #endif /* Interrupt state back up for Fixed Function only */
49
50 #if (Timer_SelLow_UsingFixedFunction)
51 /* Clear all bits but the enable bit (if it's already set) for Timer operation */
52 Timer_SelLow_CONTROL &= Timer_SelLow_CTRL_ENABLE;
53
54 /* Clear the mode bits for continuous run mode */
55 #if (CY_PSOC5A)
56 Timer_SelLow_CONTROL2 &= ((uint8)(~Timer_SelLow_CTRL_MODE_MASK));
57 #endif /* Clear bits in CONTROL2 only in PSOC5A */
58
59 #if (CY_PSOC3 || CY_PSOC5LP)
60 Timer_SelLow_CONTROL3 &= ((uint8)(~Timer_SelLow_CTRL_MODE_MASK));
61 #endif /* CONTROL3 register exists only in PSoC3 OR PSoC5LP */
62
63 /* Check if One Shot mode is enabled i.e. RunMode !=0*/
64 #if (Timer_SelLow_RunModeUsed != 0x0u)
65 /* Set 3rd bit of Control register to enable one shot mode */
66 Timer_SelLow_CONTROL |= 0x04u;
67 #endif /* One Shot enabled only when RunModeUsed is not Continuous*/
68
69 #if (Timer_SelLow_RunModeUsed == 2)
70 #if (CY_PSOC5A)
71 /* Set last 2 bits of control2 register if one shot(halt on
72 interrupt) is enabled*/
73 Timer_SelLow_CONTROL2 |= 0x03u;
74 #endif /* Set One-Shot Halt on Interrupt bit in CONTROL2 for PSoC5A */
75
76 #if (CY_PSOC3 || CY_PSOC5LP)
77 /* Set last 2 bits of control3 register if one shot(halt on
78 interrupt) is enabled*/
79 Timer_SelLow_CONTROL3 |= 0x03u;
80 #endif /* Set One-Shot Halt on Interrupt bit in CONTROL3 for PSoC3 or PSoC5LP */
81
82 #endif /* Remove section if One Shot Halt on Interrupt is not enabled */
83
84 #if (Timer_SelLow_UsingHWEnable != 0)
85 #if (CY_PSOC5A)
86 /* Set the default Run Mode of the Timer to Continuous */
87 Timer_SelLow_CONTROL2 |= Timer_SelLow_CTRL_MODE_PULSEWIDTH;
88 #endif /* Set Continuous Run Mode in CONTROL2 for PSoC5A */
89
90 #if (CY_PSOC3 || CY_PSOC5LP)
91 /* Clear and Set ROD and COD bits of CFG2 register */
92 Timer_SelLow_CONTROL3 &= ((uint8)(~Timer_SelLow_CTRL_RCOD_MASK));
93 Timer_SelLow_CONTROL3 |= Timer_SelLow_CTRL_RCOD;
94
95 /* Clear and Enable the HW enable bit in CFG2 register */
96 Timer_SelLow_CONTROL3 &= ((uint8)(~Timer_SelLow_CTRL_ENBL_MASK));
97 Timer_SelLow_CONTROL3 |= Timer_SelLow_CTRL_ENBL;
98
99 /* Set the default Run Mode of the Timer to Continuous */
100 Timer_SelLow_CONTROL3 |= Timer_SelLow_CTRL_MODE_CONTINUOUS;
101 #endif /* Set Continuous Run Mode in CONTROL3 for PSoC3ES3 or PSoC5A */
102
103 #endif /* Configure Run Mode with hardware enable */
104
105 /* Clear and Set SYNCTC and SYNCCMP bits of RT1 register */
106 Timer_SelLow_RT1 &= ((uint8)(~Timer_SelLow_RT1_MASK));
107 Timer_SelLow_RT1 |= Timer_SelLow_SYNC;
108
109 /*Enable DSI Sync all all inputs of the Timer*/
110 Timer_SelLow_RT1 &= ((uint8)(~Timer_SelLow_SYNCDSI_MASK));
111 Timer_SelLow_RT1 |= Timer_SelLow_SYNCDSI_EN;
112
113 /* Set the IRQ to use the status register interrupts */
114 Timer_SelLow_CONTROL2 |= Timer_SelLow_CTRL2_IRQ_SEL;
115 #endif /* Configuring registers of fixed function implementation */
116
117 /* Set Initial values from Configuration */
118 Timer_SelLow_WritePeriod(Timer_SelLow_INIT_PERIOD);
119 Timer_SelLow_WriteCounter(Timer_SelLow_INIT_PERIOD);
120
121 #if (Timer_SelLow_UsingHWCaptureCounter)/* Capture counter is enabled */
122 Timer_SelLow_CAPTURE_COUNT_CTRL |= Timer_SelLow_CNTR_ENABLE;
123 Timer_SelLow_SetCaptureCount(Timer_SelLow_INIT_CAPTURE_COUNT);
124 #endif /* Configure capture counter value */
125
126 #if (!Timer_SelLow_UsingFixedFunction)
127 #if (Timer_SelLow_SoftwareCaptureMode)
128 Timer_SelLow_SetCaptureMode(Timer_SelLow_INIT_CAPTURE_MODE);
129 #endif /* Set Capture Mode for UDB implementation if capture mode is software controlled */
130
131 #if (Timer_SelLow_SoftwareTriggerMode)
132 #if (!Timer_SelLow_UDB_CONTROL_REG_REMOVED)
133 if (0u == (Timer_SelLow_CONTROL & Timer_SelLow__B_TIMER__TM_SOFTWARE))
134 {
135 Timer_SelLow_SetTriggerMode(Timer_SelLow_INIT_TRIGGER_MODE);
136 }
137 #endif /* (!Timer_SelLow_UDB_CONTROL_REG_REMOVED) */
138 #endif /* Set trigger mode for UDB Implementation if trigger mode is software controlled */
139
140 /* CyEnterCriticalRegion and CyExitCriticalRegion are used to mark following region critical*/
141 /* Enter Critical Region*/
142 Timer_SelLow_interruptState = CyEnterCriticalSection();
143
144 /* Use the interrupt output of the status register for IRQ output */
145 Timer_SelLow_STATUS_AUX_CTRL |= Timer_SelLow_STATUS_ACTL_INT_EN_MASK;
146
147 /* Exit Critical Region*/
148 CyExitCriticalSection(Timer_SelLow_interruptState);
149
150 #if (Timer_SelLow_EnableTriggerMode)
151 Timer_SelLow_EnableTrigger();
152 #endif /* Set Trigger enable bit for UDB implementation in the control register*/
153
154
155 #if (Timer_SelLow_InterruptOnCaptureCount && !Timer_SelLow_UDB_CONTROL_REG_REMOVED)
156 Timer_SelLow_SetInterruptCount(Timer_SelLow_INIT_INT_CAPTURE_COUNT);
157 #endif /* Set interrupt count in UDB implementation if interrupt count feature is checked.*/
158
159 Timer_SelLow_ClearFIFO();
160 #endif /* Configure additional features of UDB implementation */
161
162 Timer_SelLow_SetInterruptMode(Timer_SelLow_INIT_INTERRUPT_MODE);
163}
