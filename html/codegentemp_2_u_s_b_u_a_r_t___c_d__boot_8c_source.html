177 /* Convert 10mS checks into 1mS checks. */

179
180 if (size > USBUART_CD_BTLDR_SIZEOF_WRITE_BUFFER)

182 size = USBUART_CD_BTLDR_SIZEOF_WRITE_BUFFER;
