1502static uint8 RTC_Main_DayOfWeek(uint8 dayOfMonth, uint8 month, uint16 year)
1503
1504{
1505 /* Calculated sequence ((31 * month) / 12) mod 7 from the Zeller's congruence */
1506 static const uint8 CYCODE RTC_Main_monthTemplate[RTC_Main_MONTHS_IN_YEAR] =
1507 {0u, 3u, 2u, 5u, 0u, 3u, 5u, 1u, 4u, 6u, 2u, 4u};
1508
1509 /* It is simpler to handle the modified year, which is year - 1 during
1510 * January and February
1511 */
1512 if(month < RTC_Main_MARCH)
1513 {
1514 year = year - 1u;
1515 }
1516
1517 /* For Gregorian calendar: d = (day + y + y/4 - y/100 + y/400 + (31*m)/12) mod 7 */
1518 return ((uint8)(((year + (((year/4u) - (year/100u)) + (year/400u))) +
1519 (((uint16)RTC_Main_monthTemplate[month - 1u]) + ((uint16) dayOfMonth))) % RTC_Main_DAYS_IN_WEEK));
1520}

1560static void RTC_Main_SetInitValues(void)
1561{
1562 uint8 i;
1563 uint8 RTC_Main_alarmCfgMaskTemp;
1564
1565 /* Clears day of month counter */
1566 RTC_Main_currentTimeDate.DayOfYear = 0u;
1567
1568 /* Increments day of year value with day in current month */
1569 RTC_Main_currentTimeDate.DayOfYear += RTC_Main_currentTimeDate.DayOfMonth;
1570
1571 /* Check leap year */
1572 if(1u == RTC_Main_LEAP_YEAR(RTC_Main_currentTimeDate.Year))
1573 {
1574 /* Set LP Year flag */
1575 RTC_Main_statusDateTime |= RTC_Main_STATUS_LY;
1576 } /* leap year flag was set */
1577 else
1578 {
1579 /* Clear LP Year flag */
1580 RTC_Main_statusDateTime &= (uint8)(~RTC_Main_STATUS_LY);
1581 } /* leap year flag was cleared */
1582
1583 /* Day of Year */
1584 for(i = 0u; i < (RTC_Main_currentTimeDate.Month - 1u); i++)
1585 {
1586 /* Increment on days in passed months */
1587 RTC_Main_currentTimeDate.DayOfYear += RTC_Main_daysInMonths[i];
1588 } /* day of year is calculated */
1589
1590 /* Leap year check */
1591 if(0u != RTC_Main_IS_BIT_SET(RTC_Main_statusDateTime, RTC_Main_STATUS_LY))
1592 {
1593 /* Leap day check */
1594 if(RTC_Main_currentTimeDate.Month > RTC_Main_FEBRUARY)
1595 {
1596 /* Add leap day */
1597 RTC_Main_currentTimeDate.DayOfYear++;
1598 } /* Do nothing for non leap day */
1599 } /* Do nothing for not leap year */
1600
1601 /* DayOfWeek */
1602 RTC_Main_currentTimeDate.DayOfWeek = RTC_Main_DayOfWeek(
1603 RTC_Main_currentTimeDate.DayOfMonth,
1604 RTC_Main_currentTimeDate.Month,
1605 RTC_Main_currentTimeDate.Year) + 1u;
1606
1607 if(RTC_Main_currentTimeDate.DayOfWeek > RTC_Main_START_OF_WEEK)
1608 {
1609 RTC_Main_currentTimeDate.DayOfWeek -= RTC_Main_START_OF_WEEK;
1610 }
1611 else
1612 {
1613 RTC_Main_currentTimeDate.DayOfWeek = RTC_Main_DAYS_IN_WEEK -
1614 (RTC_Main_START_OF_WEEK - RTC_Main_currentTimeDate.DayOfWeek);
1615 }
1616
1617 #if (1u == RTC_Main_DST_FUNC_ENABLE)
1618
1619 /* If DST values is given in a relative manner, converts to the absolute values */
1620 if(0u != RTC_Main_IS_BIT_SET(RTC_Main_dstModeType, RTC_Main_DST_RELDATE))
1621 {
1622 RTC_Main_DSTDateConversion();
1623 }
1624
1625 /* Sets DST status respect to the DST start date and time */
1626 if(RTC_Main_currentTimeDate.Month > RTC_Main_dstTimeDateStart.Month)
1627 {
1628 RTC_Main_statusDateTime |= RTC_Main_STATUS_DST;
1629 }
1630 else if(RTC_Main_currentTimeDate.Month == RTC_Main_dstTimeDateStart.Month)
1631 {
1632 if(RTC_Main_currentTimeDate.DayOfMonth > RTC_Main_dstTimeDateStart.DayOfMonth)
1633 {
1634 RTC_Main_statusDateTime |= RTC_Main_STATUS_DST;
1635 }
1636 else if(RTC_Main_currentTimeDate.DayOfMonth == RTC_Main_dstTimeDateStart.DayOfMonth)
1637 {
1638 if(RTC_Main_currentTimeDate.Hour > RTC_Main_dstTimeDateStart.Hour)
1639 {
1640 RTC_Main_statusDateTime |= RTC_Main_STATUS_DST;
1641 }
1642 }
1643 else
1644 {
1645 /* Do nothing if current day of month is less than DST stop day of month */
1646 }
1647 }
1648 else
1649 {
1650 /* Do nothing if current month is before than DST stop month */
1651 }
1652
1653 /* Clears DST status respect to the DST start date and time */
1654 if(RTC_Main_currentTimeDate.Month > RTC_Main_dstTimeDateStop.Month)
1655 {
1656 RTC_Main_statusDateTime &= (uint8)(~RTC_Main_STATUS_DST);
1657 }
