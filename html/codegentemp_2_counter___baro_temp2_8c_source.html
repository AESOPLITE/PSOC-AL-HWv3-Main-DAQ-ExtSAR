37void Counter_BaroTemp2_Init(void)
38{
39 #if (!Counter_BaroTemp2_UsingFixedFunction && !Counter_BaroTemp2_ControlRegRemoved)
40 uint8 ctrl;
41 #endif /* (!Counter_BaroTemp2_UsingFixedFunction && !Counter_BaroTemp2_ControlRegRemoved) */
42
43 #if(!Counter_BaroTemp2_UsingFixedFunction)
44 /* Interrupt State Backup for Critical Region*/
45 uint8 Counter_BaroTemp2_interruptState;
46 #endif /* (!Counter_BaroTemp2_UsingFixedFunction) */
47
48 #if (Counter_BaroTemp2_UsingFixedFunction)
49 /* Clear all bits but the enable bit (if it's already set for Timer operation */
50 Counter_BaroTemp2_CONTROL &= Counter_BaroTemp2_CTRL_ENABLE;
51
52 /* Clear the mode bits for continuous run mode */
53 #if (CY_PSOC5A)
54 Counter_BaroTemp2_CONTROL2 &= ((uint8)(~Counter_BaroTemp2_CTRL_MODE_MASK));
55 #endif /* (CY_PSOC5A) */
56 #if (CY_PSOC3 || CY_PSOC5LP)
57 Counter_BaroTemp2_CONTROL3 &= ((uint8)(~Counter_BaroTemp2_CTRL_MODE_MASK));
58 #endif /* (CY_PSOC3 || CY_PSOC5LP) */
59 /* Check if One Shot mode is enabled i.e. RunMode !=0*/
60 #if (Counter_BaroTemp2_RunModeUsed != 0x0u)
61 /* Set 3rd bit of Control register to enable one shot mode */
62 Counter_BaroTemp2_CONTROL |= Counter_BaroTemp2_ONESHOT;
63 #endif /* (Counter_BaroTemp2_RunModeUsed != 0x0u) */
64
65 /* Set the IRQ to use the status register interrupts */
66 Counter_BaroTemp2_CONTROL2 |= Counter_BaroTemp2_CTRL2_IRQ_SEL;
67
68 /* Clear and Set SYNCTC and SYNCCMP bits of RT1 register */
69 Counter_BaroTemp2_RT1 &= ((uint8)(~Counter_BaroTemp2_RT1_MASK));
70 Counter_BaroTemp2_RT1 |= Counter_BaroTemp2_SYNC;
71
72 /*Enable DSI Sync all all inputs of the Timer*/
73 Counter_BaroTemp2_RT1 &= ((uint8)(~Counter_BaroTemp2_SYNCDSI_MASK));
74 Counter_BaroTemp2_RT1 |= Counter_BaroTemp2_SYNCDSI_EN;
75
76 #else
77 #if(!Counter_BaroTemp2_ControlRegRemoved)
78 /* Set the default compare mode defined in the parameter */
79 ctrl = Counter_BaroTemp2_CONTROL & ((uint8)(~Counter_BaroTemp2_CTRL_CMPMODE_MASK));
80 Counter_BaroTemp2_CONTROL = ctrl | Counter_BaroTemp2_DEFAULT_COMPARE_MODE;
81
82 /* Set the default capture mode defined in the parameter */
83 ctrl = Counter_BaroTemp2_CONTROL & ((uint8)(~Counter_BaroTemp2_CTRL_CAPMODE_MASK));
84
85 #if( 0 != Counter_BaroTemp2_CAPTURE_MODE_CONF)
86 Counter_BaroTemp2_CONTROL = ctrl | Counter_BaroTemp2_DEFAULT_CAPTURE_MODE;
87 #else
88 Counter_BaroTemp2_CONTROL = ctrl;
89 #endif /* 0 != Counter_BaroTemp2_CAPTURE_MODE */
90
91 #endif /* (!Counter_BaroTemp2_ControlRegRemoved) */
92 #endif /* (Counter_BaroTemp2_UsingFixedFunction) */
93
94 /* Clear all data in the FIFO's */
95 #if (!Counter_BaroTemp2_UsingFixedFunction)
96 Counter_BaroTemp2_ClearFIFO();
97 #endif /* (!Counter_BaroTemp2_UsingFixedFunction) */
98
99 /* Set Initial values from Configuration */
100 Counter_BaroTemp2_WritePeriod(Counter_BaroTemp2_INIT_PERIOD_VALUE);
101 #if (!(Counter_BaroTemp2_UsingFixedFunction && (CY_PSOC5A)))
102 Counter_BaroTemp2_WriteCounter(Counter_BaroTemp2_INIT_COUNTER_VALUE);
103 #endif /* (!(Counter_BaroTemp2_UsingFixedFunction && (CY_PSOC5A))) */
104 Counter_BaroTemp2_SetInterruptMode(Counter_BaroTemp2_INIT_INTERRUPTS_MASK);
105
106 #if (!Counter_BaroTemp2_UsingFixedFunction)
107 /* Read the status register to clear the unwanted interrupts */
108 (void)Counter_BaroTemp2_ReadStatusRegister();
109 /* Set the compare value (only available to non-fixed function implementation */
110 Counter_BaroTemp2_WriteCompare(Counter_BaroTemp2_INIT_COMPARE_VALUE);
111 /* Use the interrupt output of the status register for IRQ output */
112
113 /* CyEnterCriticalRegion and CyExitCriticalRegion are used to mark following region critical*/
114 /* Enter Critical Region*/
115 Counter_BaroTemp2_interruptState = CyEnterCriticalSection();
116
117 Counter_BaroTemp2_STATUS_AUX_CTRL |= Counter_BaroTemp2_STATUS_ACTL_INT_EN_MASK;
118
119 /* Exit Critical Region*/
120 CyExitCriticalSection(Counter_BaroTemp2_interruptState);
121
122 #endif /* (!Counter_BaroTemp2_UsingFixedFunction) */
123}
