524void SPIS_Ev_WriteTxData(uint8 txData)
525{
526 #if(SPIS_Ev_TX_SOFTWARE_BUF_ENABLED)
527
528 uint8 tempStatus;
529 uint8 tmpTxBufferRead;
530
531 /* Block if buffer is full, so we don't overwrite. */
532 do
533 {
534 tmpTxBufferRead = SPIS_Ev_txBufferRead;
535 if (0u == tmpTxBufferRead)
536 {
537 tmpTxBufferRead = SPIS_Ev_TX_BUFFER_SIZE - 1u;
538 }
539 else
540 {
541 tmpTxBufferRead--;
542 }
543
544 }
545 while(tmpTxBufferRead == SPIS_Ev_txBufferWrite);
546
547 /* Disable Interrupt to protect variables that could change on interrupt. */
548 SPIS_Ev_DisableTxInt();
549
550 tempStatus = SPIS_Ev_GET_STATUS_TX(SPIS_Ev_swStatusTx);
551 SPIS_Ev_swStatusTx = tempStatus;
552
553 if((SPIS_Ev_txBufferRead == SPIS_Ev_txBufferWrite) &&
554 (0u != (SPIS_Ev_swStatusTx & SPIS_Ev_STS_TX_FIFO_NOT_FULL)))
555 {
556 /* Put data element into the TX FIFO */
557 CY_SET_REG8(SPIS_Ev_TXDATA_PTR, txData);
558 }
559 else
560 {
561 /* Put data element into the software buffer */
562 SPIS_Ev_txBufferWrite++;
563 if(SPIS_Ev_txBufferWrite >= SPIS_Ev_TX_BUFFER_SIZE)
564 {
565 SPIS_Ev_txBufferWrite = 0u;
566 }
567
568 if(SPIS_Ev_txBufferWrite == SPIS_Ev_txBufferRead)
569 {
570 SPIS_Ev_txBufferRead++;
571 if(SPIS_Ev_txBufferRead >= SPIS_Ev_RX_BUFFER_SIZE)
572 {
573 SPIS_Ev_txBufferRead = 0u;
574 }
575 SPIS_Ev_txBufferFull = 1u;
576 }
577
578 SPIS_Ev_txBuffer[SPIS_Ev_txBufferWrite] = txData;
579
580 SPIS_Ev_TX_STATUS_MASK_REG |= SPIS_Ev_STS_TX_FIFO_NOT_FULL;
581 }
582
583 /* Enable Interrupt. */
584 SPIS_Ev_EnableTxInt();
585
586 #else /* !SPIS_Ev_TX_SOFTWARE_BUF_ENABLED */
587 /* Wait until TX FIFO has a place */
588 while(0u == (SPIS_Ev_TX_STATUS_REG & SPIS_Ev_STS_TX_FIFO_NOT_FULL))
589 {
590 }
591
592 /* Put data element into the TX FIFO */
593 CY_SET_REG8(SPIS_Ev_TXDATA_PTR, txData);
594
595 #endif /* SPIS_Ev_TX_SOFTWARE_BUF_ENABLED */
596}

727uint8 SPIS_Ev_GetRxBufferSize(void)
728{
729 uint8 size;
730
731 #if(SPIS_Ev_RX_SOFTWARE_BUF_ENABLED)
732
733 /* Disable Interrupt to protect variables that could change on interrupt. */
734 SPIS_Ev_DisableRxInt();
735
736 if(SPIS_Ev_rxBufferRead == SPIS_Ev_rxBufferWrite)
737 {
738 size = 0u; /* No data in RX buffer */
739 }
740 else if(SPIS_Ev_rxBufferRead < SPIS_Ev_rxBufferWrite)
741 {
742 size = (SPIS_Ev_rxBufferWrite - SPIS_Ev_rxBufferRead);
743 }
744 else
745 {
746 size = (SPIS_Ev_RX_BUFFER_SIZE - SPIS_Ev_rxBufferRead) + SPIS_Ev_rxBufferWrite;
747 }
748
749 /* Enable interrupt. */
750 SPIS_Ev_EnableRxInt();
751
752 #else /* !SPIS_Ev_RX_SOFTWARE_BUF_ENABLED*/
753
754 /* We can only know if there is data in the fifo. */
755 size = ((SPIS_Ev_RX_STATUS_REG & SPIS_Ev_STS_RX_FIFO_NOT_EMPTY) != 0u) ? 1u : 0u;
756
757 #endif /* SPIS_Ev_RX_SOFTWARE_BUF_ENABLED */
758
759 return (size);
